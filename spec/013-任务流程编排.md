# Spec-013: 任务流程编排

## 1. 功能概述

本模块负责整个文章生成流程的状态管理和编排，包括：
- 项目状态机管理
- 各步骤间的流转控制
- 任务进度跟踪
- 错误处理和重试机制
- WebSocket实时状态推送
- 流程恢复（中断后继续）

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| 状态机 (xstate) | 流程状态管理 |
| Celery | 异步任务编排 |
| Redis | 状态缓存 |
| WebSocket | 实时推送 |

```python
# backend/app/core/state_machine.py

from enum import Enum
from typing import Dict, Set

class ProjectStatus(str, Enum):
    # 初始状态
    INIT = "init"
    
    # 搜索阶段
    SEARCHING = "searching"
    SEARCHING_COMPLETED = "searching_completed"
    
    # 过滤阶段
    FILTERING = "filtering"
    FILTERING_COMPLETED = "filtering_completed"
    
    # 知识提取阶段
    EXTRACTING = "extracting"
    EXTRACTING_COMPLETED = "extracting_completed"
    
    # 标题阶段
    TITLE_GENERATING = "title_generating"
    TITLE_GENERATING_COMPLETED = "title_generating_completed"
    TITLE_SELECTED = "title_selected"
    
    # 大纲阶段
    OUTLINE_CONFIG_SAVED = "outline_config_saved"
    OUTLINE_GENERATING = "outline_generating"
    OUTLINE_GENERATING_COMPLETED = "outline_generating_completed"
    OUTLINE_SELECTED = "outline_selected"
    
    # 文章配置阶段
    ARTICLE_CONFIG_SAVED = "article_config_saved"
    
    # 文章生成阶段
    ARTICLE_GENERATING = "article_generating"
    ARTICLE_VERIFYING = "article_verifying"
    
    # 最终状态
    COMPLETED = "completed"
    FAILED = "failed"

# 状态转换规则
TRANSITIONS: Dict[ProjectStatus, Set[ProjectStatus]] = {
    ProjectStatus.INIT: {ProjectStatus.SEARCHING, ProjectStatus.FAILED},
    ProjectStatus.SEARCHING: {ProjectStatus.SEARCHING_COMPLETED, ProjectStatus.FAILED},
    ProjectStatus.SEARCHING_COMPLETED: {ProjectStatus.FILTERING},
    ProjectStatus.FILTERING: {ProjectStatus.FILTERING_COMPLETED, ProjectStatus.FAILED},
    ProjectStatus.FILTERING_COMPLETED: {ProjectStatus.EXTRACTING},
    ProjectStatus.EXTRACTING: {ProjectStatus.EXTRACTING_COMPLETED, ProjectStatus.FAILED},
    ProjectStatus.EXTRACTING_COMPLETED: {ProjectStatus.TITLE_GENERATING},
    ProjectStatus.TITLE_GENERATING: {ProjectStatus.TITLE_GENERATING_COMPLETED, ProjectStatus.FAILED},
    ProjectStatus.TITLE_GENERATING_COMPLETED: {ProjectStatus.TITLE_SELECTED},
    ProjectStatus.TITLE_SELECTED: {ProjectStatus.OUTLINE_CONFIG_SAVED},
    ProjectStatus.OUTLINE_CONFIG_SAVED: {ProjectStatus.OUTLINE_GENERATING},
    ProjectStatus.OUTLINE_GENERATING: {ProjectStatus.OUTLINE_GENERATING_COMPLETED, ProjectStatus.FAILED},
    ProjectStatus.OUTLINE_GENERATING_COMPLETED: {ProjectStatus.OUTLINE_SELECTED},
    ProjectStatus.OUTLINE_SELECTED: {ProjectStatus.ARTICLE_CONFIG_SAVED},
    ProjectStatus.ARTICLE_CONFIG_SAVED: {ProjectStatus.ARTICLE_GENERATING},
    ProjectStatus.ARTICLE_GENERATING: {ProjectStatus.ARTICLE_VERIFYING, ProjectStatus.FAILED},
    ProjectStatus.ARTICLE_VERIFYING: {ProjectStatus.COMPLETED, ProjectStatus.FAILED},
}

def can_transition(from_status: ProjectStatus, to_status: ProjectStatus) -> bool:
    """检查状态转换是否合法"""
    return to_status in TRANSITIONS.get(from_status, set())

def get_next_status(current: ProjectStatus) -> ProjectStatus:
    """获取下一个自动状态"""
    auto_transitions = {
        ProjectStatus.SEARCHING_COMPLETED: ProjectStatus.FILTERING,
        ProjectStatus.FILTERING_COMPLETED: ProjectStatus.EXTRACTING,
        ProjectStatus.EXTRACTING_COMPLETED: ProjectStatus.TITLE_GENERATING,
        ProjectStatus.TITLE_GENERATING_COMPLETED: ProjectStatus.TITLE_SELECTED,
        ProjectStatus.TITLE_SELECTED: ProjectStatus.OUTLINE_CONFIG_SAVED,
        ProjectStatus.OUTLINE_CONFIG_SAVED: ProjectStatus.OUTLINE_GENERATING,
        ProjectStatus.OUTLINE_GENERATING_COMPLETED: ProjectStatus.OUTLINE_SELECTED,
        ProjectStatus.OUTLINE_SELECTED: ProjectStatus.ARTICLE_CONFIG_SAVED,
        ProjectStatus.ARTICLE_CONFIG_SAVED: ProjectStatus.ARTICLE_GENERATING,
    }
    return auto_transitions.get(current, current)
```

## 4. 流程编排服务

### 4.1 流程控制器

```python
# backend/app/services/workflow_service.py

from typing import Optional
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.state_machine import ProjectStatus, can_transition
from app.models.project import Project
from app.models.task import Task
from app.tasks.search_tasks import search_task
from app.tasks.filter_tasks import filter_task
from app.tasks.extract_tasks import extract_task
from app.tasks.title_tasks import generate_titles_task
from app.tasks.outline_tasks import generate_outlines_task
from app.tasks.article_tasks import generate_article_task

class WorkflowService:
    """流程编排服务"""
    
    # 每个阶段对应的任务
    STAGE_TASKS = {
        "search": search_task,
        "filter": filter_task,
        "extract": extract_task,
        "title_generate": generate_titles_task,
        "outline_generate": generate_outlines_task,
        "article_generate": generate_article_task,
    }
    
    # 每个阶段的用户交互点
    USER_INTERACTION_POINTS = {
        ProjectStatus.TITLE_GENERATING_COMPLETED,  # 需要选择标题
        ProjectStatus.OUTLINE_GENERATING_COMPLETED,  # 需要选择大纲
        ProjectStatus.OUTLINE_SELECTED,  # 需要配置文章参数
    }
    
    async def start_workflow(self, project_id: str) -> dict:
        """启动工作流"""
        async for db in self._get_session():
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError(f"Project {project_id} not found")
            
            # 开始搜索
            await self._transition_to(db, project, ProjectStatus.SEARCHING)
            celery_result = search_task.delay(project_id)
            
            await self._create_task_record(db, project_id, "search", celery_result.id)
            
            return {
                "project_id": project_id,
                "status": ProjectStatus.SEARCHING,
                "task_id": celery_result.id
            }
    
    async def advance_stage(self, project_id: str, stage: str) -> dict:
        """推进到下一阶段"""
        async for db in self._get_session():
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError(f"Project {project_id} not found")
            
            current_status = ProjectStatus(project.status)
            
            # 获取阶段对应的状态转换
            stage_status_map = {
                "search": (ProjectStatus.SEARCHING, ProjectStatus.SEARCHING_COMPLETED),
                "filter": (ProjectStatus.FILTERING, ProjectStatus.FILTERING_COMPLETED),
                "extract": (ProjectStatus.EXTRACTING, ProjectStatus.EXTRACTING_COMPLETED),
                "title_generate": (ProjectStatus.TITLE_GENERATING, ProjectStatus.TITLE_GENERATING_COMPLETED),
                "outline_generate": (ProjectStatus.OUTLINE_GENERATING, ProjectStatus.OUTLINE_GENERATING_COMPLETED),
                "article_generate": (ProjectStatus.ARTICLE_GENERATING, ProjectStatus.ARTICLE_VERIFYING),
            }
            
            if stage not in stage_status_map:
                raise ValueError(f"Unknown stage: {stage}")
            
            start_status, end_status = stage_status_map[stage]
            
            # 启动异步任务
            task_func = self.STAGE_TASKS.get(stage)
            if not task_func:
                raise ValueError(f"No task for stage: {stage}")
            
            await self._transition_to(db, project, start_status)
            celery_result = task_func.delay(project_id)
            await self._create_task_record(db, project_id, stage, celery_result.id)
            
            return {
                "project_id": project_id,
                "status": start_status,
                "task_id": celery_result.id
            }
    
    async def handle_task_completion(self, project_id: str, task_type: str, success: bool) -> None:
        """处理任务完成"""
        async for db in self._get_session():
            project = await db.get(Project, project_id)
            if not project:
                return
            
            current_status = ProjectStatus(project.status)
            
            if not success:
                await self._transition_to(db, project, ProjectStatus.FAILED)
                return
            
            # 根据任务类型更新状态
            completion_map = {
                "search": ProjectStatus.SEARCHING_COMPLETED,
                "filter": ProjectStatus.FILTERING_COMPLETED,
                "extract": ProjectStatus.EXTRACTING_COMPLETED,
                "title_generate": ProjectStatus.TITLE_GENERATING_COMPLETED,
                "outline_generate": ProjectStatus.OUTLINE_GENERATING_COMPLETED,
                "article_generate": ProjectStatus.ARTICLE_VERIFYING,
            }
            
            next_status = completion_map.get(task_type)
            if next_status:
                await self._transition_to(db, project, next_status)
    
    async def retry_failed_task(self, project_id: str) -> dict:
        """重试失败的任务"""
        async for db in self._get_session():
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError("Project not found")
            
            # 获取失败的任务
            result = await db.execute(
                select(Task).where(
                    Task.project_id == project_id,
                    Task.status == "failed"
                ).order_by(Task.created_at.desc())
            )
            failed_task = result.scalar_one_or_none()
            
            if not failed_task:
                raise ValueError("No failed task found")
            
            # 重新启动任务
            task_func = self.STAGE_TASKS.get(failed_task.task_type)
            if task_func:
                celery_result = task_func.delay(project_id)
                failed_task.celery_task_id = celery_result.id
                failed_task.status = "pending"
                await db.commit()
                
                return {"task_id": celery_result.id, "status": "retrying"}
    
    async def _transition_to(self, db: AsyncSession, project: Project, new_status: ProjectStatus) -> None:
        """状态转换"""
        current_status = ProjectStatus(project.status)
        
        if not can_transition(current_status, new_status):
            raise ValueError(f"Invalid transition: {current_status} -> {new_status}")
        
        project.status = new_status.value
        await db.commit()
    
    async def _create_task_record(self, db: AsyncSession, project_id: str, task_type: str, celery_task_id: str) -> Task:
        """创建任务记录"""
        task = Task(
            project_id=project_id,
            task_type=task_type,
            celery_task_id=celery_task_id,
            status="pending"
        )
        db.add(task)
        await db.commit()
        await db.refresh(task)
        return task
    
    async def _get_session(self):
        from app.database import async_session_factory
        return async_session_factory()

workflow_service = WorkflowService()
```

## 5. 错误处理与重试

### 5.1 重试策略

```python
# backend/app/core/retry.py

from typing import Callable, Any
import asyncio
from functools import wraps

class RetryConfig:
    """重试配置"""
    MAX_RETRIES = 3
    BASE_DELAY = 1.0  # 秒
    MAX_DELAY = 60.0
    EXPONENTIAL_BASE = 2

def with_retry(
    max_retries: int = RetryConfig.MAX_RETRIES,
    base_delay: float = RetryConfig.BASE_DELAY,
    exceptions: tuple = (Exception,)
):
    """重试装饰器"""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        async def wrapper(*args, **kwargs) -> Any:
            last_exception = None
            
            for attempt in range(max_retries + 1):
                try:
                    return await func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    
                    if attempt < max_retries:
                        delay = min(
                            base_delay * (RetryConfig.EXPONENTIAL_BASE ** attempt),
                            RetryConfig.MAX_DELAY
                        )
                        print(f"Retry {attempt + 1}/{max_retries} after {delay}s: {e}")
                        await asyncio.sleep(delay)
                    else:
                        print(f"All {max_retries} retries failed: {e}")
            
            raise last_exception
        
        return wrapper
    return decorator
```

### 6.1 连接管理器

```python
# backend/app/core/websocket.py

from fastapi import WebSocket
from typing import Dict, Set
import json
import asyncio

class ConnectionManager:
    def __init__(self):
        # project_id -> set of websockets
        self.active_connections: Dict[str, Set[WebSocket]] = {}
    
    async def connect(self, project_id: str, websocket: WebSocket):
        await websocket.accept()
        if project_id not in self.active_connections:
            self.active_connections[project_id] = set()
        self.active_connections[project_id].add(websocket)
    
    def disconnect(self, project_id: str, websocket: WebSocket):
        if project_id in self.active_connections:
            self.active_connections[project_id].discard(websocket)
            if not self.active_connections[project_id]:
                del self.active_connections[project_id]
    
    async def send_to_project(self, project_id: str, message: dict):
        if project_id in self.active_connections:
            disconnected = set()
            for ws in self.active_connections[project_id]:
                try:
                    await ws.send_json(message)
                except Exception:
                    disconnected.add(ws)
            # 清理断开的连接
            for ws in disconnected:
                self.active_connections[project_id].discard(ws)

manager = ConnectionManager()

# 消息类型
class MessageType:
    PROGRESS = "progress"
    STATUS_CHANGE = "status_change"
    TASK_COMPLETE = "task_complete"
    TASK_FAILED = "task_failed"
    ERROR = "error"

async def notify_progress(project_id: str, progress: int, message: str, stage: str = None):
    await manager.send_to_project(project_id, {
        "type": MessageType.PROGRESS,
        "progress": progress,
        "message": message,
        "stage": stage,
    })

async def notify_status_change(project_id: str, old_status: str, new_status: str):
    await manager.send_to_project(project_id, {
        "type": MessageType.STATUS_CHANGE,
        "old_status": old_status,
        "new_status": new_status,
    })

async def notify_task_complete(project_id: str, task_type: str, result: dict):
    await manager.send_to_project(project_id, {
        "type": MessageType.TASK_COMPLETE,
        "task_type": task_type,
        "result": result,
    })

async def notify_task_failed(project_id: str, task_type: str, error: str):
    await manager.send_to_project(project_id, {
        "type": MessageType.TASK_FAILED,
        "task_type": task_type,
        "error": error,
    })
```

```python
# backend/app/services/recovery_service.py

from sqlalchemy import select
from app.models.project import Project
from app.models.task import Task
from app.core.state_machine import ProjectStatus

class RecoveryService:
    """流程恢复服务"""
    
    # 状态到恢复点的映射
    RECOVERY_POINTS = {
        ProjectStatus.SEARCHING_COMPLETED: "filter",
        ProjectStatus.FILTERING_COMPLETED: "extract",
        ProjectStatus.EXTRACTING_COMPLETED: "title_generate",
        ProjectStatus.TITLE_GENERATING_COMPLETED: "title_select",  # 需要用户交互
        ProjectStatus.TITLE_SELECTED: "outline_config",  # 需要用户交互
        ProjectStatus.OUTLINE_CONFIG_SAVED: "outline_generate",
        ProjectStatus.OUTLINE_GENERATING_COMPLETED: "outline_select",  # 需要用户交互
        ProjectStatus.OUTLINE_SELECTED: "article_config",  # 需要用户交互
        ProjectStatus.ARTICLE_CONFIG_SAVED: "article_generate",
    }
    
    async def get_recovery_point(self, project_id: str) -> dict:
        """获取项目的恢复点"""
        async for db in self._get_session():
            project = await db.get(Project, project_id)
            if not project:
                return None
            
            current_status = ProjectStatus(project.status)
            recovery_point = self.RECOVERY_POINTS.get(current_status)
            
            # 检查是否有正在运行的任务
            running_task = await db.execute(
                select(Task).where(
                    Task.project_id == project_id,
                    Task.status == "running"
                )
            )
            task = running_task.scalar_one_or_none()
            
            return {
                "project_id": project_id,
                "current_status": current_status,
                "recovery_point": recovery_point,
                "has_running_task": task is not None,
                "next_action": self._get_next_action(current_status),
            }
    
    def _get_next_action(self, status: ProjectStatus) -> str:
        """获取下一步操作"""
        actions = {
            ProjectStatus.INIT: "start_search",
            ProjectStatus.SEARCHING: "wait",
            ProjectStatus.SEARCHING_COMPLETED: "start_filter",
            ProjectStatus.FILTERING: "wait",
            ProjectStatus.FILTERING_COMPLETED: "start_extract",

            ProjectStatus.EXTRACTING_COMPLETED: "start_title_generate",

            ProjectStatus.COMPLETED: "view_article",

        }
        return actions.get(status, "unknown")
    
    async def _get_session(self):
        from app.database import async_session_factory
        return async_session_factory()
```

## 8. 前端流程管理Hook

```typescript
// frontend/src/hooks/useWorkflow.ts

import { useEffect, useState, useCallback, useRef } from 'react';
import { useProjectStore } from '@/stores/projectStore';

interface WorkflowState {
  currentStage: string;
  progress: number;
  message: string;
  status: string;
}

export function useWorkflow(projectId: string) {
  const [state, setState] = useState<WorkflowState>({
    currentStage: 'init',
    progress: 0,
    message: '',
    status: 'idle',
  });
  const wsRef = useRef<WebSocket | null>(null);

  useEffect(() => {
    // 建立WebSocket连接
    const ws = new WebSocket(`${process.env.NEXT_PUBLIC_WS_URL}/tasks/${projectId}`);
    wsRef.current = ws;

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      
      switch (data.type) {
        case 'progress':
          setState(prev => ({
            ...prev,
            progress: data.progress,
            message: data.message,
            currentStage: data.stage || prev.currentStage,
          }));
          break;
        
        case 'status_change':
          setState(prev => ({
            ...prev,
            status: data.new_status,
          }));
          break;
        
        case 'task_complete':
          setState(prev => ({
            ...prev,
            progress: 100,
            message: `${data.task_type} 完成`,
          }));
          break;
        
        case 'task_failed':
          setState(prev => ({
            ...prev,
            status: 'failed',
            message: data.error,
          }));
          break;
      }
    };

    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };

    ws.onclose = () => {
      console.log('WebSocket closed');
    };

    return () => {
      ws.close();
    };
  }, [projectId]);

  const advanceStage = useCallback(async (stage: string) => {
    // 调用API推进阶段
    const response = await fetch(`/api/projects/${projectId}/workflow/advance`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ stage }),
    });
    return response.json();
  }, [projectId]);

  return {
    ...state,
    advanceStage,
  };
}
```
