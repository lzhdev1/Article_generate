# Spec-011: 文章生成模块

## 1. 功能概述

本模块负责根据用户选择的大纲和配置，使用LLM生成完整的文章内容。生成过程中需要进行内容验证：
- 可靠性验证：事实、数据是否准确
- 时效性验证：信息是否过时
- 真实性验证：是否存在虚假信息

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| 阿里云百炼 (Qwen-Max) | 文章生成 |
| Bing Search API | 内容验证搜索 |
| Celery | 异步任务 |
| WebSocket | 实时进度推送 |

---

## 3. 流程设计

```
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│  构建生成Prompt  │────▶│  逐章节生成     │────▶│   内容整合       │
│  (大纲+知识)    │     │  (LLM生成)       │     │                 │
└─────────────────┘     └─────────────────┘     └─────────────────┘
                                                         │
                                                         ▼
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│  生成附加内容    │◀────│  内容验证       │◀────│   生成完成       │
│  (总结/FAQ/图)  │     │  (网络搜索验证)  │     │                 │
└─────────────────┘     └─────────────────┘     └─────────────────┘
```

### 3.1 验证策略

| 验证类型 | 方法 | 阈值 |
|----------|------|------|
| 可靠性 | 搜索验证关键事实 | 至少1个可靠来源 |
| 时效性 | 检查数据发布时间 | 优先使用近2年数据 |
| 真实性 | 交叉验证关键声明 | 至少2个来源一致 |

## 4. 数据模型

### 4.1 文章表

```sql
CREATE TABLE articles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID UNIQUE NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    
    title VARCHAR(300) NOT NULL,
    content TEXT NOT NULL,           -- Markdown格式正文
    content_html TEXT,               -- HTML格式正文
    
    word_count INTEGER,
    
    -- 附加内容
    summary TEXT,                    -- 要点总结
    faq JSONB DEFAULT '[]',          -- FAQ列表
    images JSONB DEFAULT '[]',       -- 图片信息
    videos JSONB DEFAULT '[]',       -- 视频信息
    
    -- 状态
    status VARCHAR(50) NOT NULL DEFAULT 'generating',
    -- status: generating, verifying, completed, failed
    
    -- 元数据
    metadata JSONB DEFAULT '{}',
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

### 4.2 验证记录表

```sql
CREATE TABLE verifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    article_id UUID NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
    
    claim TEXT NOT NULL,             -- 需要验证的声明
    is_verified BOOLEAN,            -- 是否验证通过
    confidence_score DECIMAL(3,2),  -- 置信度
    evidence JSONB DEFAULT '[]',    -- 验证证据
    verification_type VARCHAR(50),  -- fact_check/timeliness/authenticity
    
    verified_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_verifications_article_id ON verifications(article_id);
```

## 6. 核心代码实现

### 6.1 文章生成服务

```python
# backend/app/services/article_service.py

from typing import List, Dict
from sqlalchemy import select
from datetime import datetime

from app.services.llm_service import llm_service
from app.services.search_service import search_service
from app.services.extract_service import extract_service
from app.models.outline import Outline, OutlineConfig
from app.models.knowledge import ExtractedKnowledge
from app.models.title import Title
from app.models.article import Article, ArticleConfig, Verification
from app.models.project import Project

class ArticleService:
    
    SECTION_PROMPT = """请根据以下要求撰写文章的章节内容。

## 文章标题
{title}

## 文章主题
{topic}

## 当前章节信息
- 章节标题: {section_heading}
- 章节描述: {section_description}
- 子话题: {subtopics}
- 目标字数: {word_count}字

## 写作要求
- 可读性: {readability}
- 写作风格: {tone_of_voice}
- 人称视角: {point_of_view}
- 额外要求: {additional_requirements}

## 参考知识
{reference_knowledge}

## 上下文
前文概要: {previous_context}

请以Markdown格式输出本章节内容，确保：
1. 内容与大纲描述一致
2. 覆盖所有子话题
3. 字数约{word_count}字
4. 与前后文自然衔接
5. 使用Markdown标题、列表等格式"""

    VERIFY_PROMPT = """请验证以下文章中的关键声明是否准确。

## 待验证的文章片段
{content}

请识别文章中的关键事实性声明（数据、日期、引用、专业论断等），并对每条声明进行验证。

对每条声明：
1. 明确声明内容
2. 判断是否可靠（基于你的知识）
3. 给出置信度(0-1)
4. 说明判断依据

以JSON格式返回：
{{
  "claims": [
    {{
      "claim": "声明内容",
      "type": "fact_check|timeliness|authenticity",
      "is_verified": true/false,
      "confidence": 0.85,
      "reasoning": "判断依据"
    }}
  ]
}}"""

    SUMMARY_PROMPT = """请为以下文章生成一个简洁的要点总结，用3-5个要点概括文章核心内容。

## 文章标题
{title}

## 文章内容
{content}

请以Markdown列表格式输出：
- 要点1
- 要点2
- ..."""

    FAQ_PROMPT = """请根据文章内容生成{count}个常见问题及解答。

## 文章标题
{title}

## 文章内容摘要
{content_summary}

问题应涵盖：
1. 读者可能感兴趣的核心问题
2. 文章中没有详细展开但读者可能想知道的
3. 实际应用中可能遇到的问题

以JSON格式返回：
{{
  "faq": [
    {{"question": "问题", "answer": "答案"}}
  ]
}}"""

    async def generate_article(self, project_id: str) -> Dict:
        """生成完整文章"""
        async for db in self._get_session():
            # 获取项目数据
            project = await db.get(Project, project_id)
            title = await self._get_selected_title(db, project_id)
            outline = await self._get_selected_outline(db, project_id)
            outline_config = await self._get_outline_config(db, project_id)
            article_config = await self._get_article_config(db, project_id)
            knowledge = await self._get_knowledge(db, project_id)
            
            # 创建文章记录
            article = Article(
                project_id=project_id,
                title=title.content,
                content="",
                status="generating"
            )
            db.add(article)
            await db.commit()
            await db.refresh(article)
            
            # 逐章节生成
            sections_content = []
            previous_context = ""
            
            for i, section in enumerate(outline.content.get("sections", [])):
                # 构建章节内容
                section_content = await self._generate_section(
                    title=title.content,
                    topic=project.topic,
                    section=section,
                    config=outline_config,
                    knowledge=knowledge,
                    previous_context=previous_context
                )
                
                sections_content.append(section_content)
                previous_context = section_content[:500]  # 用作下一章节上下文
            
            # 整合内容
            full_content = self._assemble_content(title.content, outline.content, sections_content)
            
            # 生成附加内容
            summary = None
            faq = []
            
            if article_config.add_summary:
                summary = await self._generate_summary(title.content, full_content)
            
            if article_config.add_faq:
                faq = await self._generate_faq(title.content, full_content, article_config.faq_count)
            
            # 更新文章
            article.content = full_content
            article.summary = summary
            article.faq = faq
            article.word_count = len(full_content)
            article.status = "verifying"
            
            project.status = "article_verifying"
            await db.commit()
            
            # 内容验证
            verifications = await self._verify_content(full_content)
            
            # 保存验证记录
            for v in verifications:
                verification = Verification(
                    article_id=article.id,
                    claim=v.get("claim", ""),
                    is_verified=v.get("is_verified"),
                    confidence_score=v.get("confidence"),
                    verification_type=v.get("type"),
                    evidence=[{"reasoning": v.get("reasoning", "")}],
                    verified_at=datetime.utcnow()
                )
                db.add(verification)
            
            # 更新状态
            article.status = "completed"
            article.content_html = self._markdown_to_html(full_content)
            project.status = "article_completed"
            await db.commit()
            
            return {
                "article_id": str(article.id),
                "word_count": article.word_count,
                "verifications_count": len(verifications)
            }
    
    async def _generate_section(self, title, topic, section, config, knowledge, previous_context) -> str:
        """生成单个章节"""
        # 构建参考知识
        knowledge_text = ""
        for item in knowledge:
            for kp in (item.key_points or [])[:5]:
                if isinstance(kp, dict):
                    knowledge_text += f"- {kp.get('content', '')}\n"
        
        prompt = self.SECTION_PROMPT.format(
            title=title,
            topic=topic,
            section_heading=section.get("heading", ""),
            section_description=section.get("description", ""),
            subtopics=", ".join(section.get("subtopics", [])),
            word_count=section.get("word_count", 300),
            readability=config.readability,
            tone_of_voice=config.tone_of_voice,
            point_of_view=config.point_of_view,
            additional_requirements=config.additional_requirements or "无",
            reference_knowledge=knowledge_text[:3000],
            previous_context=previous_context[:1000]
        )
        
        response = await llm_service.chat(prompt)
        return response
    
    def _assemble_content(self, title: str, outline: Dict, sections: List[str]) -> str:
        """组装完整内容"""
        content = f"# {title}\n\n"
        
        outline_sections = outline.get("sections", [])
        for i, section in enumerate(outline_sections):
            if i < len(sections):
                content += f"## {section.get('heading', '')}\n\n"
                content += sections[i] + "\n\n"
        
        return content.strip()
    
    async def _generate_summary(self, title: str, content: str) -> str:
        """生成要点总结"""
        prompt = self.SUMMARY_PROMPT.format(
            title=title,
            content=content[:5000]
        )
        response = await llm_service.chat(prompt)
        return response
    
    async def _generate_faq(self, title: str, content: str, count: int) -> List[Dict]:
        """生成FAQ"""
        prompt = self.FAQ_PROMPT.format(
            count=count,
            title=title,
            content_summary=content[:3000]
        )
        response = await llm_service.chat(prompt)
        result = llm_service.parse_json(response)
        return result.get("faq", [])
    
    async def _verify_content(self, content: str) -> List[Dict]:
        """验证内容"""
        # 分批验证（每2000字一批）
        chunks = [content[i:i+2000] for i in range(0, len(content), 2000)]
        all_claims = []
        
        for chunk in chunks:
            prompt = self.VERIFY_PROMPT.format(content=chunk)
            response = await llm_service.chat(prompt)
            result = llm_service.parse_json(response)
            claims = result.get("claims", [])
            all_claims.extend(claims)
        
        return all_claims
    
    def _markdown_to_html(self, markdown: str) -> str:
        """Markdown转HTML"""
        import markdown
        return markdown.markdown(markdown, extensions=['extra', 'codehilite'])
    
    async def _get_session(self):
        from app.database import async_session_factory
        return async_session_factory()
    
    async def _get_selected_title(self, db, project_id):
        result = await db.execute(
            select(Title).where(Title.project_id == project_id, Title.is_selected == True)
        )
        return result.scalar_one()
    
    async def _get_selected_outline(self, db, project_id):
        result = await db.execute(
            select(Outline).where(Outline.project_id == project_id, Outline.is_selected == True)
        )
        return result.scalar_one()
    
    async def _get_outline_config(self, db, project_id):
        result = await db.execute(
            select(OutlineConfig).where(OutlineConfig.project_id == project_id)
        )
        return result.scalar_one()
    
    async def _get_article_configs(self, db, project_id):
        result = await db.execute(
            select(ArticleConfig).where(ArticleConfig.project_id == project_id)
        )
        return result.scalar_one()
    
    async def _get_knowledge(self, db, project_id):
        result = await db.execute(
            select(ExtractedKnowledge).where(ExtractedKnowledge.project_id == project_id)
        )
        return result.scalars().all()

article_service = ArticleService()
```

```python
# backend/app/tasks/article_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.task import Task
from app.services.article_service import article_service

@celery.task(bind=True, name="app.tasks.generate_article")
def generate_article_task(self, project_id: str):
    return asyncio.run(_generate_article_async(self, project_id))

async def _generate_article_async(celery_task, project_id: str):
    async for db in async_session_factory():
        try:
            task = await _get_or_create_task(db, project_id, "article_generate", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            await _update_progress(db, task, 10, "正在准备生成...")
            await _update_progress(db, task, 20, "正在生成文章章节...")
            
            result = await article_service.generate_article(project_id)
            
            await _update_progress(db, task, 90, "正在完成...")
            
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = result
            await db.commit()
            
            return result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

# 辅助函数（略）
async def _get_or_create_task(db, project_id, task_type, celery_task_id):
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    if not task:
        task = Task(project_id=project_id, task_type=task_type, celery_task_id=celery_task_id)
        db.add(task)
        await db.flush()
    return task

async def _update_progress(db, task, progress, message=None):
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```
