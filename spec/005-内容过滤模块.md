# Spec-005: 内容过滤模块

## 1. 功能概述

本模块负责对搜索结果进行智能过滤，筛选出与用户Topic高度相关的内容。使用大模型对每条搜索结果进行相关性评估和评分。

核心功能：
- 批量评估搜索结果与Topic的相关性
- 为每条结果打分（0-1分）
- 标记是否相关（阈值过滤）
- 过滤结果展示页面

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| 阿里云百炼 (Qwen-Max) | 内容相关性评估 |
| Celery | 异步任务处理 |
| Redis | 任务队列 |

## 3. 流程设计

```
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│  获取搜索结果    │────▶│  批量评估        │────▶│   评分排序       │
│  (有内容的)     │     │  (LLM打分)       │     │                 │
└─────────────────┘     └─────────────────┘     └─────────────────┘
                                                         │
                                                         ▼
                                                ┌─────────────────┐
                                                │   阈值过滤       │
                                                │ (score >= 0.6)  │
                                                └─────────────────┘
```

### 3.1 过滤策略

1. **预筛选**：只处理有crawled_content的结果
2. **批量评估**：每次发送5-10条给LLM评估
3. **评分标准**：
   - 0.8-1.0：高度相关，核心参考资料
   - 0.6-0.8：相关，可作为补充
   - 0.4-0.6：部分相关，可选
   - < 0.4：不相关，过滤掉
4. **结果限制**：保留top 10-15条相关内容

## 6. 核心代码实现

### 6.1 过滤服务

```python
# backend/app/services/filter_service.py

from typing import List, Tuple
from app.services.llm_service import llm_service
from app.models.search import SearchItem

class FilterService:
    
    FILTER_PROMPT = """你是一个专业的内容分析助手。请评估以下搜索结果与用户主题的相关性。

用户主题: {topic}

请对以下每篇文章进行评估，给出相关性评分(0-1之间的小数)和简要原因。

评分标准:
- 0.8-1.0: 高度相关，文章核心内容与主题直接相关
- 0.6-0.8: 相关，文章内容与主题有较多关联
- 0.4-0.6: 部分相关，文章只是部分涉及主题
- 0.2-0.4: 关联较少，只是间接提及
- 0-0.2: 不相关

搜索结果:
{articles}

请以JSON数组格式返回评估结果:
[
  {{"index": 0, "score": 0.85, "reason": "原因说明"}},
  ...
]"""

    async def filter_articles(
        self, 
        topic: str, 
        articles: List[SearchItem],
        batch_size: int = 5
    ) -> List[Tuple[SearchItem, float, str]]:
        """批量评估文章相关性"""
        results = []
        
        # 分批处理
        for i in range(0, len(articles), batch_size):
            batch = articles[i:i + batch_size]
            batch_results = await self._evaluate_batch(topic, batch)
            results.extend(batch_results)
        
        return results
    
    async def _evaluate_batch(
        self, 
        topic: str, 
        articles: List[SearchItem]
    ) -> List[Tuple[SearchItem, float, str]]:
        """评估一批文章"""
        # 构建评估文本
        articles_text = ""
        for idx, article in enumerate(articles):
            content_preview = (article.crawled_content or article.snippet or "")[:1000]
            articles_text += f"""
---文章 {idx}---
标题: {article.title}
内容摘要: {content_preview}
"""
        
        prompt = self.FILTER_PROMPT.format(
            topic=topic,
            articles=articles_text
        )
        
        # 调用LLM
        response = await llm_service.chat(prompt)
        
        # 解析响应
        evaluations = llm_service.parse_json_array(response)
        
        results = []
        for eval_item in evaluations:
            idx = eval_item.get("index", 0)
            score = float(eval_item.get("score", 0))
            reason = eval_item.get("reason", "")
            
            if 0 <= idx < len(articles):
                results.append((articles[idx], score, reason))
        
        return results
    
    async def filter_and_update(
        self,
        project_id: str,
        topic: str,
        articles: List[SearchItem],
        threshold: float = 0.6,
        max_results: int = 15
    ) -> List[SearchItem]:
        """过滤并更新数据库"""
        from sqlalchemy.ext.asyncio import AsyncSession
        from app.database import async_session_factory
        from datetime import datetime
        
        # 评估所有文章
        evaluations = await self.filter_articles(topic, articles)
        
        # 按分数排序
        evaluations.sort(key=lambda x: x[1], reverse=True)
        
        # 取前N条
        relevant_items = []
        async for db in async_session_factory():
            for article, score, reason in evaluations[:max_results]:
                if score >= threshold:
                    article.is_relevant = True
                    article.relevance_score = score
                    article.filter_reason = reason
                    article.filtered_at = datetime.utcnow()
                    db.add(article)
                    relevant_items.append(article)
                else:
                    article.is_relevant = False
                    article.relevance_score = score
                    article.filter_reason = reason
                    article.filtered_at = datetime.utcnow()
                    db.add(article)
            
            await db.commit()
        
        return relevant_items

filter_service = FilterService()
```

### 6.2 过滤任务

```python
# backend/app/tasks/filter_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.search import SearchItem
from app.models.task import Task
from app.models.project import Project
from app.services.filter_service import filter_service

@celery.task(bind=True, name="app.tasks.filter")
def filter_task(self, project_id: str, threshold: float = 0.6, max_results: int = 15):
    """执行内容过滤任务"""
    return asyncio.run(_filter_task_async(self, project_id, threshold, max_results))

async def _filter_task_async(celery_task, project_id: str, threshold: float, max_results: int):
    async for db in async_session_factory():
        try:
            # 获取项目信息
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError(f"Project {project_id} not found")
            
            # 获取任务记录
            task = await _get_or_create_task(db, project_id, "filter", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            # 获取搜索结果（只取有内容的）
            results = await db.execute(
                select(SearchItem).where(
                    SearchItem.project_id == project_id,
                    SearchItem.crawled_content.isnot(None)
                )
            )
            articles = results.scalars().all()
            
            if not articles:
                task.status = "failed"
                task.error = "没有找到可过滤的搜索结果"
                await db.commit()
                return
            
            await _update_progress(db, task, 20, f"开始过滤 {len(articles)} 条搜索结果...")
            
            # 执行过滤
            relevant_items = await filter_service.filter_and_update(
                project_id=project_id,
                topic=project.topic,
                articles=articles,
                threshold=threshold,
                max_results=max_results
            )
            
            await _update_progress(db, task, 90, f"过滤完成，保留 {len(relevant_items)} 条相关内容")
            
            # 更新项目状态
            project.status = "filtering_completed"
            await db.commit()
            
            # 完成任务
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = {
                "total_processed": len(articles),
                "relevant_count": len(relevant_items),
                "threshold": threshold,
            }
            await db.commit()
            
            return task.result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

# 辅助函数（与search_tasks.py相同）
async def _get_or_create_task(db, project_id, task_type, celery_task_id):
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    if not task:
        task = Task(project_id=project_id, task_type=task_type, celery_task_id=celery_task_id)
        db.add(task)
        await db.flush()
    return task

async def _update_progress(db, task, progress, message=None):
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```

### 6.3 API端点

```python
# backend/app/api/endpoints/filter.py

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from pydantic import BaseModel, Field

from app.database import get_db
from app.models.search import SearchItem
from app.tasks.filter_tasks import filter_task

router = APIRouter()

class FilterRequest(BaseModel):
    threshold: float = Field(default=0.6, ge=0, le=1, description="相关性阈值")
    max_results: int = Field(default=15, ge=1, le=50, description="最大结果数")

@router.post("/projects/{project_id}/filter")
async def start_filter(
    project_id: str,
    request: FilterRequest = FilterRequest(),
    db: AsyncSession = Depends(get_db)
):
    """启动过滤任务"""
    project = await db.get(Project, project_id)
    if not project:
        raise HTTPException(status_code=404, detail="Project not found")
    
    celery_result = filter_task.delay(project_id, request.threshold, request.max_results)
    
    return {
        "task_id": celery_result.id,
        "status": "pending"
    }

@router.get("/projects/{project_id}/filter/results")
async def get_filter_results(
    project_id: str,
    db: AsyncSession = Depends(get_db)
):
    """获取过滤结果"""
    # 获取相关内容
    results = await db.execute(
        select(SearchItem)
        .where(
            SearchItem.project_id == project_id,
            SearchItem.is_relevant == True
        )
        .order_by(SearchItem.relevance_score.desc())
    )
    relevant_items = results.scalars().all()
    
    # 获取总数
    total_results = await db.execute(
        select(SearchItem).where(
            SearchItem.project_id == project_id,
            SearchItem.is_relevant.isnot(None)
        )
    )
    total_count = len(total_results.scalars().all())
    
    return {
        "relevant_items": [
            {
                "id": str(item.id),
                "title": item.title,
                "url": item.url,
                "snippet": item.snippet,
                "source": item.source,
                "relevance_score": item.relevance_score,
                "relevance_reason": item.filter_reason,
            }
            for item in relevant_items
        ],
        "relevant_count": len(relevant_items),
        "total_count": total_count
    }
```
