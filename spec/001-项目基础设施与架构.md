### 2.3 AI框架架构说明

```
┌─────────────────────────────────────────────────────────────┐
│                   LangGraph (工作流编排层)                    │
│                                                              │
│  StateGraph: 定义整个文章生成流程                              │
│  ├── 节点: search → filter → extract → title → outline      │
│  │         → config → generate → verify → output             │
│  ├── 条件边: verify失败 → 重新生成（循环）                    │
│  ├── interrupt: 标题选择、大纲选择（Human-in-the-Loop）       │
│  └── Checkpointer: PostgreSQL持久化状态，支持中断恢复          │
└─────────────────────────────────────────────────────────────┘
                              │
                              │ 调用
                              ▼
┌─────────────────────────────────────────────────────────────┐
│                   LangChain (AI组件层)                        │
│                                                              │
│  ├── ChatOpenAI: 统一LLM调用接口（兼容百炼）                  │
│  ├── PromptTemplate: Prompt模板管理                           │
│  ├── Document Loaders: 网页内容加载                           │
│  ├── Text Splitters: 文本智能切分                             │
│  ├── Embeddings: 文本向量化（DashScope Embedding）            │
│  ├── VectorStore: 知识向量存储（PGVector）                    │
│  ├── Retriever: 知识语义检索                                  │
│  └── Output Parser: LLM输出结构化解析                         │
└─────────────────────────────────────────────────────────────┘
                              │
                              │ 调度
                              ▼
┌─────────────────────────────────────────────────────────────┐
│                   Celery (底层任务分发层)                      │
│                                                              │
│  ├── 大规模并行任务（批量爬取、批量Embedding）                 │
│  ├── 长时间运行任务（文章生成可能需要数分钟）                   │
│  └── 与LangGraph互补：LangGraph编排智能决策，                 │
│      Celery负责底层异步执行                                    │
└─────────────────────────────────────────────────────────────┘
```

**为什么同时用 LangGraph + Celery？**
- **LangGraph**: 负责有状态的工作流编排（循环、条件分支、人工交互点、状态持久化）
- **Celery**: 负责底层重计算任务（批量爬取、批量Embedding）和分布式任务分发
- 两者互补，LangGraph通过自定义Node调用Celery任务

### 4.2 数据表定义

#### 4.2.1 projects 表 - 项目主表

```sql
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic VARCHAR(500) NOT NULL,
    status VARCHAR(50) NOT NULL DEFAULT 'init',
    -- status: init, searching, filtering, extracting, title_generating, 
    --         title_selecting, outline_config, outline_generating, 
    --         outline_selecting, final_config, generating, verifying, completed, failed
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_projects_status ON projects(status);
CREATE INDEX idx_projects_created_at ON projects(created_at DESC);
```

#### 4.2.2 search_items 表 - 搜索结果

```sql
CREATE TABLE search_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    title VARCHAR(500) NOT NULL,
    url TEXT NOT NULL,
    snippet TEXT,
    source VARCHAR(100),
    published_date DATE,
    crawled_content TEXT,
    is_relevant BOOLEAN DEFAULT NULL,
    relevance_score DECIMAL(3,2) DEFAULT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_search_items_project_id ON search_items(project_id);
CREATE INDEX idx_search_items_is_relevant ON search_items(is_relevant) WHERE is_relevant IS NOT NULL;
```

#### 4.2.3 extracted_knowledge 表 - 提取的知识

```sql
CREATE TABLE extracted_knowledge (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    search_item_id UUID REFERENCES search_items(id) ON DELETE SET NULL,
    key_points JSONB NOT NULL DEFAULT '[]',
    summary TEXT,
    facts JSONB NOT NULL DEFAULT '[]',
    quotes JSONB NOT NULL DEFAULT '[]',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_extracted_knowledge_project_id ON extracted_knowledge(project_id);
```

#### 4.2.3 titles 表 - 生成的标题

```sql
CREATE TABLE titles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    content VARCHAR(300) NOT NULL,
    reasoning TEXT,
    is_selected BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_titles_project_id ON titles(project_id);
```

#### 4.2.4 outline_configs 表 - 大纲配置

```sql
CREATE TABLE outline_configs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID UNIQUE NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    word_count INTEGER NOT NULL DEFAULT 2000,
    readability VARCHAR(50) NOT NULL DEFAULT 'general',
    -- readability: general, easy, moderate, advanced, expert
    tone_of_voice VARCHAR(50) NOT NULL DEFAULT 'professional',
    -- tone: casual, conversational, professional, formal, humorous, authoritative
    point_of_view VARCHAR(50) NOT NULL DEFAULT 'third_person',
    -- pov: first_person, second_person, third_person
    additional_requirements TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

#### 4.2.5 outlines 表 - 生成的大纲

```sql
CREATE TABLE outlines (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    content JSONB NOT NULL,
    -- content结构: [{title: string, description: string, subtopics: [...]}]
    reasoning TEXT,
    is_selected BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_outlines_project_id ON outlines(project_id);
```

#### 4.2.6 article_configs 表 - 文章最终配置

```sql
CREATE TABLE article_configs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID UNIQUE NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    add_images BOOLEAN DEFAULT TRUE,
    add_videos BOOLEAN DEFAULT FALSE,
    add_summary BOOLEAN DEFAULT TRUE,
    add_faq BOOLEAN DEFAULT TRUE,
    image_count INTEGER DEFAULT 3,
    faq_count INTEGER DEFAULT 5,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

#### 4.2.7 articles 表 - 生成的文章

```sql
CREATE TABLE articles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID UNIQUE NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    title VARCHAR(300) NOT NULL,
    content TEXT NOT NULL,
    content_html TEXT,
    word_count INTEGER,
    status VARCHAR(50) NOT NULL DEFAULT 'generating',
    -- status: generating, verifying, completed, failed
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
```

#### 4.2.8 verifications 表 - 内容验证记录

```sql
CREATE TABLE verifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    article_id UUID NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
    claim TEXT NOT NULL,
    is_verified BOOLEAN,
    confidence_score DECIMAL(3,2),
    evidence JSONB DEFAULT '[]',
    verification_type VARCHAR(50),
    -- type: fact_check, timeliness, authenticity
    verified_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_verifications_article_id ON verifications(article_id);
```

import { create } from 'zustand';

interface ProjectState {

  titles: Title[];
  outlineConfig: OutlineConfig | null;
  outlines: Outline[];
  articleConfig: ArticleConfig | null;
  article: Article | null;
  
  // 任务状态
  currentTask: TaskStatus | null;
  isLoading: boolean;
  error: string | null;
  
  // Actions
  setProject: (project: Project) => void;
  setSearchResults: (results: SearchItem[]) => void;
  setExtractedKnowledge: (knowledge: ExtractedKnowledge[]) => void;
  setTitles: (titles: Title[]) => void;
  selectTitle: (titleId: string) => void;
  setOutlineConfig: (config: OutlineConfig) => void;
  setOutlines: (outlines: Outline[]) => void;
  selectOutline: (outlineId: string) => void;
  setArticleConfig: (config: ArticleConfig) => void;
  setArticle: (article: Article) => void;
  setCurrentTask: (task: TaskStatus) => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;
  reset: () => void;
}

export const useProjectStore = create<ProjectState>((set) => ({

  titles: [],
  outlineConfig: null,
  outlines: [],
  articleConfig: null,
  article: null,
  currentTask: null,

  setProject: (project) => set({ currentProject: project }),

from contextlib import asynccontextmanager

@asynccontextmanager
async def lifespan(app: FastAPI):
    # Startup

        await conn.run_sync(Base.metadata.create_all)
    yield
    # Shutdown

app = FastAPI(
    title="Article Generator API",

    version="1.0.0",

    CORSMiddleware,

    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
