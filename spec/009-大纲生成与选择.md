# Spec-009: 大纲生成与选择

## 1. 功能概述

本模块负责：
1. 使用LLM分析生成标题时引用的文章内容
2. 提取分析每篇文章的内容和风格
3. 根据用户配置的需求，生成至少2个大纲供用户选择
4. 用户选择大纲后进入最终配置页面

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| 阿里云百炼 (Qwen-Max) | 大纲生成 |
| Celery | 异步任务 |
| Framer Motion | 卡片动画 |

---

## 3. 数据模型

### 3.1 大纲表

```sql
CREATE TABLE outlines (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    
    -- 大纲内容 (JSON结构)
    content JSONB NOT NULL,
    -- 结构:
    -- {
    --   "title": "大纲标题",
    --   "sections": [
    --     {
    --       "heading": "章节标题",
    --       "description": "章节描述",
    --       "subtopics": ["子话题1", "子话题2"],
    --       "word_count": 500
    --     }
    --   ],
    --   "total_word_count": 2000
    -- }
    
    reasoning TEXT,          -- 生成思路说明
    style VARCHAR(100),      -- 大纲风格/结构类型
    is_selected BOOLEAN DEFAULT FALSE,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_outlines_project_id ON outlines(project_id);
```

## 5. 核心代码实现

### 5.1 大纲生成服务

```python
# backend/app/services/outline_service.py

from typing import List, Dict
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.services.llm_service import llm_service
from app.models.knowledge import ExtractedKnowledge
from app.models.outline import OutlineConfig, Outline
from app.models.search import SearchItem
from app.models.title import Title
from app.models.project import Project

class OutlineService:
    
    GENERATE_PROMPT = """你是一位专业的内容策划师，擅长设计清晰、有逻辑的文章结构。

请根据以下信息，生成{count}个不同风格的文章大纲。

## 文章标题
{title}

## 文章主题
{topic}

## 用户需求配置
- 目标字数: {word_count}字
- 可读性: {readability}
- 写作风格: {tone_of_voice}
- 人称视角: {point_of_view}
- 额外要求: {additional_requirements}

## 参考知识摘要

### 关键论点
{key_points}

### 事实数据
{facts}

### 参考文章的风格分析
{style_analysis}

## 要求
请生成{count}个不同结构风格的大纲，每个大纲包含：
- 完整的章节结构（至少4-6个章节）
- 每章节的标题和描述
- 每章节的子话题
- 预估字数分配

大纲风格参考：
1. comprehensive: 全面系统型 - 覆盖所有方面
2. problem_solution: 问题-解决型 - 从痛点出发
3. step_by_step: 循序渐进型 - 按步骤展开
4. comparison: 对比分析型 - 多角度对比
5. case_study: 案例分析型 - 以案例为主线

以JSON格式返回：
{{
  "outlines": [
    {{
      "content": {{
        "title": "大纲标题",
        "sections": [
          {{
            "heading": "章节标题",
            "description": "章节描述",
            "subtopics": ["子话题1", "子话题2"],
            "word_count": 300
          }}
        ],
        "total_word_count": {word_count}
      }},
      "reasoning": "设计思路说明",
      "style": "style_name"
    }}
  ]
}}"""

    STYLE_ANALYSIS_PROMPT = """请分析以下参考文章的内容特点和写作风格：

{articles}

请从以下维度进行分析：
1. 结构特点：文章如何组织内容
2. 写作风格：语言特点和表达方式
3. 论证方式：如何引用数据和案例
4. 读者互动：如何吸引和保持读者兴趣

以简洁的要点形式返回分析结果。"""

    async def analyze_reference_styles(self, articles: List[SearchItem]) -> str:
        """分析参考文章的风格"""
        articles_text = ""
        for i, article in enumerate(articles[:5]):  # 最多分析5篇
            content = (article.crawled_content or article.snippet or "")[:1500]
            articles_text += f"\n---文章{i+1}: {article.title}---\n{content}\n"
        
        prompt = self.STYLE_ANALYSIS_PROMPT.format(articles=articles_text)
        response = await llm_service.chat(prompt)
        return response
    
    async def generate_outlines(
        self, 
        project_id: str,
        count: int = 3
    ) -> List[Dict]:
        """生成大纲"""
        async for db in self._get_session():
            # 获取项目信息
            project = await db.get(Project, project_id)
            
            # 获取选中的标题
            title_result = await db.execute(
                select(Title).where(Title.project_id == project_id, Title.is_selected == True)
            )
            selected_title = title_result.scalar_one()
            
            # 获取配置
            config_result = await db.execute(
                select(OutlineConfig).where(OutlineConfig.project_id == project_id)
            )
            config = config_result.scalar_one()
            
            # 获取知识库
            knowledge_result = await db.execute(
                select(ExtractedKnowledge).where(ExtractedKnowledge.project_id == project_id)
            )
            knowledge_items = knowledge_result.scalars().all()
            
            # 获取参考文章
            articles_result = await db.execute(
                select(SearchItem).where(
                    SearchItem.project_id == project_id,
                    SearchItem.is_relevant == True
                ).limit(5)
            )
            articles = articles_result.scalars().all()
            
            # 分析文章风格
            style_analysis = await self.analyze_reference_styles(articles)
            
            # 整合知识
            key_points = []
            facts = []
            for item in knowledge_items:
                for kp in (item.key_points or [])[:5]:
                    if isinstance(kp, dict):
                        key_points.append(kp.get("content", ""))
                for fact in (item.facts or [])[:3]:
                    if isinstance(fact, dict):
                        facts.append(fact.get("content", ""))
            
            # 构建prompt
            prompt = self.GENERATE_PROMPT.format(
                count=count,
                title=selected_title.content,
                topic=project.topic,
                word_count=config.word_count,
                readability=config.readability,
                tone_of_voice=config.tone_of_voice,
                point_of_view=config.point_of_view,
                additional_requirements=config.additional_requirements or "无",
                key_points="\n".join(f"- {kp}" for kp in key_points[:15]),
                facts="\n".join(f"- {f}" for f in facts[:10]),
                style_analysis=style_analysis
            )
            
            # 调用LLM
            response = await llm_service.chat(prompt)
            result = llm_service.parse_json(response)
            
            return result.get("outlines", [])
    
    async def save_outlines(self, project_id: str, outlines_data: List[Dict]) -> List[Outline]:
        """保存大纲"""
        async for db in self._get_session():
            saved = []
            for data in outlines_data:
                outline = Outline(
                    project_id=project_id,
                    content=data["content"],
                    reasoning=data.get("reasoning", ""),
                    style=data.get("style", "general")
                )
                db.add(outline)
                saved.append(outline)
            
            await db.commit()
            for o in saved:
                await db.refresh(o)
            return saved
    
    async def select_outline(self, project_id: str, outline_id: str) -> Outline:
        """选择大纲"""
        async for db in self._get_session():
            # 取消所有选择
            results = await db.execute(
                select(Outline).where(Outline.project_id == project_id)
            )
            for o in results.scalars().all():
                o.is_selected = False
            
            # 设置新选择
            outline = await db.get(Outline, outline_id)
            outline.is_selected = True
            
            # 更新项目状态
            project = await db.get(Project, project_id)
            project.status = "outline_selected"
            
            await db.commit()
            return outline
    
    async def _get_session(self):
        from app.database import async_session_factory
        return async_session_factory()

outline_service = OutlineService()
```

### 5.2 大纲生成任务

```python
# backend/app/tasks/outline_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.task import Task
from app.services.outline_service import outline_service

@celery.task(bind=True, name="app.tasks.generate_outlines")
def generate_outlines_task(self, project_id: str, count: int = 3):
    return asyncio.run(_generate_outlines_async(self, project_id, count))

async def _generate_outlines_async(celery_task, project_id: str, count: int):
    async for db in async_session_factory():
        try:
            task = await _get_or_create_task(db, project_id, "outline_generate", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            await _update_progress(db, task, 20, "正在分析参考文章风格...")
            await _update_progress(db, task, 50, "正在生成大纲...")
            
            outlines_data = await outline_service.generate_outlines(project_id, count)
            
            await _update_progress(db, task, 80, "正在保存大纲...")
            
            outlines = await outline_service.save_outlines(project_id, outlines_data)
            
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = {"count": len(outlines)}
            await db.commit()
            
            return task.result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

# 辅助函数（略）
async def _get_or_create_task(db, project_id, task_type, celery_task_id):
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    if not task:
        task = Task(project_id=project_id, task_type=task_type, celery_task_id=celery_task_id)
        db.add(task)
        await db.flush()
    return task

async def _update_progress(db, task, progress, message=None):
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```

```tsx
"use client"

import { useEffect, useState } from "react"
import { useParams, useRouter } from "next/navigation"
import { motion } from "framer-motion"
import { Check, ChevronDown, ChevronUp, Loader2, FileText } from "lucide-react"

import { Button } from "@/components/ui/button"

interface OutlineSection {

  description: string
  subtopics: string[]
  word_count: number
}

interface OutlineContent {
  title: string
  sections: OutlineSection[]
  total_word_count: number
}

interface Outline {
  id: string
  content: OutlineContent
  reasoning: string
  style: string
  is_selected: boolean
}

const styleLabels: Record<string, string> = {
  comprehensive: "全面系统型",
  problem_solution: "问题-解决型",
  step_by_step: "循序渐进型",
  comparison: "对比分析型",
  case_study: "案例分析型",
  general: "通用型",
}

export default function OutlinesPage() {
  const params = useParams()
  const router = useRouter()
  const [outlines, setOutlines] = useState<Outline[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [isGenerating, setIsGenerating] = useState(false)
  const [selectingId, setSelectingId] = useState<string | null>(null)
  const [expandedId, setExpandedId] = useState<string | null>(null)

  useEffect(() => {
    loadOutlines()
  }, [])

  const loadOutlines = async () => {
    try {
      const data = await api.outline.list(params.id as string)
      if (data.outlines.length > 0) {
        setOutlines(data.outlines)
      } else {
        await generateOutlines()
      }
    } catch (error) {
      console.error(error)
    } finally {
      setIsLoading(false)
    }
  }

  const generateOutlines = async () => {
    setIsGenerating(true)
    try {
      await api.outline.generate(params.id as string)
      // 轮询等待
      await new Promise<void>((resolve) => {
        const interval = setInterval(async () => {
          const data = await api.outline.list(params.id as string)
          if (data.outlines.length > 0) {
            clearInterval(interval)
            setOutlines(data.outlines)
            resolve()
          }
        }, 2000)
      })
    } catch (error) {
      console.error(error)
    } finally {
      setIsGenerating(false)
    }
  }

    setSelectingId(outlineId)
    try {
      await api.outline.select(params.id as string, outlineId)
      setOutlines(outlines.map(o => ({
        ...o,
        is_selected: o.id === outlineId
      })))
      setTimeout(() => {
        router.push(`/project/${params.id}/config`)
      }, 500)
    } catch (error) {
      console.error(error)
      setSelectingId(null)
    }
  }

  if (isLoading) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <LoadingSpinner size="lg" />
      </div>
    )
  }

    { id: "search", label: "搜索", status: "completed" },
    { id: "filter", label: "过滤", status: "completed" },
    { id: "extract", label: "提取", status: "completed" },
    { id: "title", label: "标题", status: "completed" },
    { id: "outline", label: "大纲", status: "current" },
    { id: "config", label: "配置", status: "upcoming" },
    { id: "generate", label: "生成", status: "upcoming" },
  ]

  return (
    <div className="min-h-screen bg-background">
      <div className="mx-auto max-w-6xl px-4 py-8">
        <StepIndicator steps={steps} className="mb-8" />
        
        <div className="mb-8 text-center">
          <h1 className="text-3xl font-bold">请选择一个大纲</h1>
          <p className="mt-2 text-neutral-500">
            为您生成了{outlines.length}个不同风格的大纲方案
          </p>
        </div>

        {isGenerating ? (
          <div className="flex flex-col items-center justify-center py-20">
            <LoadingSpinner size="lg" />
            <p className="mt-4 text-neutral-500">正在生成大纲...</p>
          </div>
        ) : (

            "grid gap-6",
            outlines.length === 2 ? "md:grid-cols-2" : "md:grid-cols-2 lg:grid-cols-3"
          )}>
            {outlines.map((outline, index) => (
              <motion.div
                key={outline.id}
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: index * 0.1 }}
                className={outlines.length === 3 && index === 2 ? "md:col-span-2 lg:col-span-1" : ""}
              >
                <Card className={cn(
                  "h-full flex flex-col transition-all duration-300",
                  outline.is_selected
                    ? "border-primary-500 bg-primary-50 shadow-glow-primary"
                    : "hover:border-primary-200 hover:shadow-medium"
                )}>
                  <CardHeader>
                    <div className="flex items-center justify-between">
                      <CardTitle className="text-lg">
                        方案 {String.fromCharCode(65 + index)}
                      </CardTitle>
                      <span className="rounded-full bg-neutral-100 px-3 py-1 text-xs font-medium">
                        {styleLabels[outline.style] || outline.style}
                      </span>
                    </div>
                  </CardHeader>
                  
                  <CardContent className="flex-1">
                    {/* 章节列表 */}
                    <div className="space-y-3">
                      {outline.content.sections.map((section, sIndex) => (
                        <div key={sIndex} className="flex items-start gap-3">
                          <FileText className="mt-0.5 h-4 w-4 shrink-0 text-primary-500" />
                          <div className="flex-1">
                            <p className="text-sm font-medium text-neutral-900">
                              {section.heading}
                            </p>
                            <p className="text-xs text-neutral-500">
                              约{section.word_count}字
                            </p>
                          </div>
                        </div>

                    </div>
                    
                    {/* 展开详情 */}
                    <button

                      {expandedId === outline.id ? (

                        <>查看详情 <ChevronDown className="h-4 w-4" /></>
                      )}
                    </button>

                        initial={{ height: 0, opacity: 0 }}

                        className="mt-3 space-y-3 border-t pt-3"

                        {outline.content.sections.map((section, sIndex) => (

                          <p className="text-xs text-neutral-500">{outline.reasoning}</p>
                        </div>
                      </motion.div>
                    )}
                  </CardContent>
                  
                  <CardFooter>
                    <Button
                      className="w-full"

                    >
                      {selectingId === outline.id ? (

                      ) : outline.is_selected ? (
                        <>
                          <Check className="mr-2 h-4 w-4" />
                          已选择
                        </>
                      ) : (
                        "选择此大纲"
                      )}
                    </Button>
                  </CardFooter>
                </Card>
              </motion.div>
            ))}
          </div>
        )}
      </div>
    </div>
  )
}
```
