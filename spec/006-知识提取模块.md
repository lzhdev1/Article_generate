# Spec-006: 知识提取模块

## 1. 功能概述

本模块负责对过滤后的相关内容进行深度知识提取，从文章中提取：
- 关键论点/观点
- 事实数据
- 引用语句
- 核心摘要

提取的知识将作为后续标题生成、大纲生成、文章生成的参考依据。

### 3.1 提取内容类型

| 类型 | 说明 | 示例 |
|------|------|------|
| key_points | 关键论点/观点 | "AI可将个性化学习效率提升40%" |
| facts | 事实数据 | "2024年全球AI教育市场规模达$20B" |
| quotes | 引用语句 | 原文中值得引用的精彩语句 |
| summary | 文章摘要 | 200字以内的核心内容概括 |

```sql
CREATE TABLE extracted_knowledge (

    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,

    quotes JSONB NOT NULL DEFAULT '[]',       -- 引用语句数组
    summary TEXT,                              -- 内容摘要

    source_title VARCHAR(500),
    source_url TEXT,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_extracted_knowledge_project_id ON extracted_knowledge(project_id);
```

## 6. 核心代码实现

### 6.1 知识提取服务

```python
# backend/app/services/extract_service.py

from typing import List, Dict, Any
from app.services.llm_service import llm_service
from app.models.search import SearchItem

class ExtractService:
    
    EXTRACT_PROMPT = """请从以下文章中提取关键知识信息。

文章标题: {title}
文章URL: {url}
文章内容:
{content}

请提取以下类型的信息：

1. 关键论点 (key_points): 文章的主要观点和结论
   - content: 论点内容
   - importance: 重要程度 (high/medium/low)
   - category: 分类（如：技术、应用、趋势、挑战等）

2. 事实数据 (facts): 具体的数据、日期、定义等
   - content: 事实内容
   - type: 类型 (statistic/date/definition/other)
   - source: 数据来源（如果提到）

3. 引用语句 (quotes): 值得引用的精彩语句
   - content: 引用内容
   - author: 作者（如果有）
   - context: 上下文简述

4. 摘要 (summary): 200字以内的核心内容概括

请以JSON格式返回：
{{
  "key_points": [
    {{"content": "...", "importance": "high", "category": "..."}}
  ],
  "facts": [
    {{"content": "...", "type": "statistic", "source": "..."}}
  ],
  "quotes": [
    {{"content": "...", "author": "...", "context": "..."}}
  ],
  "summary": "..."
}}"""

    async def extract_from_article(self, article: SearchItem) -> Dict[str, Any]:
        """从单篇文章提取知识"""
        content = article.crawled_content or article.snippet or ""
        
        # 截断过长内容
        if len(content) > 5000:
            content = content[:5000] + "..."
        
        prompt = self.EXTRACT_PROMPT.format(
            title=article.title,
            url=article.url,
            content=content
        )
        
        response = await llm_service.chat(prompt)
        result = llm_service.parse_json(response)
        
        return result
    
    async def extract_from_articles(
        self, 
        articles: List[SearchItem]
    ) -> List[Dict[str, Any]]:
        """从多篇文章批量提取知识"""
        results = []
        
        for article in articles:
            try:
                knowledge = await self.extract_from_article(article)
                knowledge["source_title"] = article.title
                knowledge["source_url"] = article.url
                knowledge["search_item_id"] = str(article.id)
                results.append(knowledge)
            except Exception as e:
                print(f"Failed to extract from {article.title}: {e}")
                continue
        
        return results
    
    async def integrate_knowledge(
        self, 
        knowledge_items: List[Dict[str, Any]]
    ) -> Dict[str, Any]:
        """整合所有知识，去重和归类"""
        all_key_points = []
        all_facts = []
        all_quotes = []
        
        for item in knowledge_items:
            all_key_points.extend(item.get("key_points", []))
            all_facts.extend(item.get("facts", []))
            all_quotes.extend(item.get("quotes", []))
        
        # 简单去重（基于内容相似度）
        unique_key_points = self._deduplicate(all_key_points, "content")
        unique_facts = self._deduplicate(all_facts, "content")
        unique_quotes = self._deduplicate(all_quotes, "content")
        
        return {
            "key_points": unique_key_points,
            "facts": unique_facts,
            "quotes": unique_quotes,
            "statistics": {
                "total_articles": len(knowledge_items),
                "total_key_points": len(unique_key_points),
                "total_facts": len(unique_facts),
                "total_quotes": len(unique_quotes),
            }
        }
    
    def _deduplicate(self, items: List[Dict], key: str) -> List[Dict]:
        """简单去重"""
        seen = set()
        unique = []
        for item in items:
            content = item.get(key, "")
            # 简单的前50字符匹配去重
            content_key = content[:50].lower()
            if content_key not in seen:
                seen.add(content_key)
                unique.append(item)
        return unique

extract_service = ExtractService()
```

### 6.2 知识提取任务

```python
# backend/app/tasks/extract_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.search import SearchItem
from app.models.knowledge import ExtractedKnowledge
from app.models.task import Task
from app.models.project import Project
from app.services.extract_service import extract_service

@celery.task(bind=True, name="app.tasks.extract")
def extract_task(self, project_id: str):
    """执行知识提取任务"""
    return asyncio.run(_extract_task_async(self, project_id))

async def _extract_task_async(celery_task, project_id: str):
    async for db in async_session_factory():
        try:
            # 获取项目信息
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError(f"Project {project_id} not found")
            
            # 获取任务记录
            task = await _get_or_create_task(db, project_id, "extract", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            # 获取相关内容
            results = await db.execute(
                select(SearchItem).where(
                    SearchItem.project_id == project_id,
                    SearchItem.is_relevant == True
                )
            )
            articles = results.scalars().all()
            
            if not articles:
                task.status = "failed"
                task.error = "没有相关内容可供提取"
                await db.commit()
                return
            
            await _update_progress(db, task, 10, f"开始从 {len(articles)} 篇文章提取知识...")
            
            # 逐篇提取
            knowledge_items = []
            for i, article in enumerate(articles):
                progress = 10 + int((i / len(articles)) * 70)
                await _update_progress(db, task, progress, f"正在提取: {article.title[:30]}...")
                
                knowledge = await extract_service.extract_from_article(article)
                
                # 保存到数据库
                extracted = ExtractedKnowledge(
                    project_id=project_id,
                    search_item_id=article.id,
                    key_points=knowledge.get("key_points", []),
                    facts=knowledge.get("facts", []),
                    quotes=knowledge.get("quotes", []),
                    summary=knowledge.get("summary", ""),
                    source_title=article.title,
                    source_url=article.url
                )
                db.add(extracted)
                knowledge_items.append(knowledge)
                
                await asyncio.sleep(0.5)  # 控制请求频率
            
            await db.commit()
            
            await _update_progress(db, task, 85, "正在整合知识...")
            
            # 整合知识
            integrated = await extract_service.integrate_knowledge(knowledge_items)
            
            # 更新项目状态
            project.status = "extracting_completed"
            project.metadata = {**(project.metadata or {}), "knowledge_stats": integrated["statistics"]}
            await db.commit()
            
            # 完成任务
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = integrated["statistics"]
            await db.commit()
            
            return task.result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

# 辅助函数（与其他tasks类似）
async def _get_or_create_task(db, project_id, task_type, celery_task_id):
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    if not task:
        task = Task(project_id=project_id, task_type=task_type, celery_task_id=celery_task_id)
        db.add(task)
        await db.flush()
    return task

async def _update_progress(db, task, progress, message=None):
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```
