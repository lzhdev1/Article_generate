# Spec-012: 文章展示页面

## 1. 功能概述

文章生成完成后，本页面负责展示最终生成的文章，包括：
- 文章标题和正文（Markdown渲染）
- 要点总结
- FAQ常见问题
- 内容验证状态
- 文章导出功能

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| react-markdown | Markdown渲染 |
| remark-gfm | GFM扩展支持 |
| rehype-highlight | 代码高亮 |
| html2pdf.js | PDF导出 |
| Framer Motion | 动画 |

---

## 3. API设计

### 3.1 获取完整文章

```
GET /api/projects/{project_id}/article
```

## 5. 前端组件

### 5.1 文章展示页面

```tsx
// frontend/src/app/project/[id]/article/page.tsx

"use client"

import { useEffect, useState } from "react"
import { useParams, useRouter } from "next/navigation"
import { motion } from "framer-motion"
import ReactMarkdown from "react-markdown"
import remarkGfm from "remark-gfm"
import rehypeHighlight from "rehype-highlight"
import {
  ArrowLeft, FileText, Clock, Shield, Download, Copy,
  CheckCircle, AlertCircle, ChevronDown, ChevronUp,
  ListChecks, HelpCircle, Sparkles
} from "lucide-react"

import { Button } from "@/components/ui/button"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { LoadingSpinner } from "@/components/shared/loading-spinner"
import { api } from "@/services/api"
import { cn } from "@/lib/utils"

interface Article {
  id: string
  title: string
  content: string
  content_html: string
  word_count: number
  status: string
  summary?: string
  faq?: { question: string; answer: string }[]
  verifications?: {
    total: number
    verified: number
    unverified: number
    confidence_score: number
  }
}

export default function ArticlePage() {
  const params = useParams()
  const router = useRouter()
  const [article, setArticle] = useState<Article | null>(null)
  const [isLoading, setIsLoading] = useState(true)
  const [faqExpanded, setFaqExpanded] = useState<Record<number, boolean>>({})
  const [showSummary, setShowSummary] = useState(true)
  const [showVerifications, setShowVerifications] = useState(true)

  useEffect(() => {
    loadArticle()
  }, [])

  const loadArticle = async () => {
    try {
      const data = await api.article.get(params.id as string)
      setArticle(data)
    } catch (error) {
      console.error(error)
    } finally {
      setIsLoading(false)
    }
  }

  const handleCopy = async () => {
    if (!article) return
    await navigator.clipboard.writeText(article.content)
    // 显示toast
  }

  const handleExport = async (format: 'markdown' | 'pdf') => {
    if (!article) return
    
    if (format === 'markdown') {
      const blob = new Blob([article.content], { type: 'text/markdown' })
      const url = URL.createObjectURL(blob)
      const a = document.createElement('a')
      a.href = url
      a.download = `${article.title}.md`
      a.click()
      URL.revokeObjectURL(url)
    }
  }

  if (isLoading) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <LoadingSpinner size="lg" />
      </div>
    )
  }

  if (!article) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <p className="text-neutral-500">文章未找到</p>
      </div>
    )
  }

  const readingTime = Math.ceil(article.word_count / 250)
  const verificationRate = article.verifications
    ? Math.round((article.verifications.verified / article.verifications.total) * 100)
    : 0

  return (
    <div className="min-h-screen bg-background">
      {/* 顶部操作栏 */}
      <div className="sticky top-0 z-10 border-b border-neutral-100 bg-white/80 backdrop-blur-sm">
        <div className="mx-auto flex max-w-4xl items-center justify-between px-4 py-3">
          <Button
            variant="ghost"
            onClick={() => router.push("/")}
          >
            <ArrowLeft className="mr-2 h-4 w-4" />
            返回首页
          </Button>
          <div className="flex items-center gap-2">
            <Button variant="outline" size="sm" onClick={() => handleCopy()}>
              <Copy className="mr-1 h-3 w-3" />
              复制
            </Button>
            <Button variant="outline" size="sm" onClick={() => handleExport('markdown')}>
              <Download className="mr-1 h-3 w-3" />
              Markdown
            </Button>
            <Button variant="outline" size="sm" onClick={() => handleExport('pdf')}>
              <Download className="mr-1 h-3 w-3" />
              PDF
            </Button>
          </div>
        </div>
      </div>

      <div className="mx-auto max-w-4xl px-4 py-8">
        {/* 文章头部 */}
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          className="mb-8"
        >
          <h1 className="text-4xl font-bold leading-tight text-neutral-900">
            {article.title}
          </h1>
          
          <div className="mt-4 flex flex-wrap items-center gap-4 text-sm text-neutral-500">
            <span className="flex items-center gap-1">
              <FileText className="h-4 w-4" />
              {article.word_count.toLocaleString()}字
            </span>
            <span className="flex items-center gap-1">
              <Clock className="h-4 w-4" />
              约{readingTime}分钟阅读
            </span>
            {article.verifications && (
              <span className="flex items-center gap-1">
                <Shield className="h-4 w-4" />
                验证通过: {verificationRate}%
              </span>
            )}
          </div>
        </motion.div>

        {/* 要点总结 */}
        {article.summary && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.1 }}
            className="mb-6"
          >
            <Card className="bg-gradient-to-r from-primary-50 to-accent-50">
              <CardHeader className="pb-3">
                <div
                  className="flex cursor-pointer items-center justify-between"
                  onClick={() => setShowSummary(!showSummary)}
                >
                  <CardTitle className="flex items-center gap-2 text-base">
                    <ListChecks className="h-5 w-5 text-primary-500" />
                    要点总结
                  </CardTitle>
                  {showSummary ? (
                    <ChevronUp className="h-4 w-4 text-neutral-400" />
                  ) : (
                    <ChevronDown className="h-4 w-4 text-neutral-400" />
                  )}
                </div>
              </CardHeader>
              {showSummary && (
                <CardContent>
                  <div className="prose prose-sm max-w-none text-neutral-700">
                    <ReactMarkdown remarkPlugins={[remarkGfm]}>
                      {article.summary}
                    </ReactMarkdown>
                  </div>
                </CardContent>
              )}
            </Card>
          </motion.div>
        )}

        {/* 文章正文 */}
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: 0.2 }}
          className="mb-8"
        >
          <Card>
            <CardContent className="p-8">
              <div className="prose prose-lg max-w-none prose-headings:text-neutral-900 prose-p:text-neutral-700 prose-a:text-primary-600">
                <ReactMarkdown
                  remarkPlugins={[remarkGfm]}
                  rehypePlugins={[rehypeHighlight]}
                >
                  {article.content}
                </ReactMarkdown>
              </div>
            </CardContent>
          </Card>
        </motion.div>

        {/* FAQ */}
        {article.faq && article.faq.length > 0 && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.3 }}
            className="mb-6"
          >
            <Card>
              <CardHeader>
                <CardTitle className="flex items-center gap-2 text-lg">
                  <HelpCircle className="h-5 w-5 text-orange-500" />
                  常见问题 (FAQ)
                </CardTitle>
              </CardHeader>
              <CardContent className="space-y-4">
                {article.faq.map((item, index) => (
                  <div key={index} className="border-b border-neutral-100 pb-4 last:border-0">
                    <div
                      className="flex cursor-pointer items-start justify-between"
                      onClick={() => setFaqExpanded(prev => ({ ...prev, [index]: !prev[index] }))}
                    >
                      <div className="flex-1">
                        <p className="font-medium text-neutral-900">
                          Q: {item.question}
                        </p>
                        {faqExpanded[index] && (
                          <motion.div
                            initial={{ height: 0, opacity: 0 }}
                            animate={{ height: "auto", opacity: 1 }}
                            className="mt-2"
                          >
                            <p className="text-neutral-600">A: {item.answer}</p>
                          </motion.div>
                        )}
                      </div>
                      {faqExpanded[index] ? (
                        <ChevronUp className="ml-2 h-4 w-4 shrink-0 text-neutral-400" />
                      ) : (
                        <ChevronDown className="ml-2 h-4 w-4 shrink-0 text-neutral-400" />
                      )}
                    </div>
                  </div>
                ))}
              </CardContent>
            </Card>
          </motion.div>
        )}

        {/* 验证报告 */}
        {article.verifications && (
          <motion.div
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ delay: 0.4 }}
            className="mb-8"
          >
            <Card>
              <CardHeader className="pb-3">
                <div
                  className="flex cursor-pointer items-center justify-between"
                  onClick={() => setShowVerifications(!showVerifications)}
                >
                  <CardTitle className="flex items-center gap-2 text-base">
                    <Shield className="h-5 w-5 text-accent-500" />
                    内容验证报告
                  </CardTitle>
                  {showVerifications ? (
                    <ChevronUp className="h-4 w-4 text-neutral-400" />
                  ) : (
                    <ChevronDown className="h-4 w-4 text-neutral-400" />
                  )}
                </div>
              </CardHeader>
              {showVerifications && (
                <CardContent>
                  <div className="mb-4 grid grid-cols-3 gap-4">
                    <div className="rounded-lg bg-neutral-50 p-3 text-center">
                      <p className="text-2xl font-bold text-neutral-900">
                        {article.verifications.total}
                      </p>
                      <p className="text-xs text-neutral-500">总验证项</p>
                    </div>
                    <div className="rounded-lg bg-green-50 p-3 text-center">
                      <p className="text-2xl font-bold text-green-600">
                        {article.verifications.verified}
                      </p>
                      <p className="text-xs text-green-600">通过</p>
                    </div>
                    <div className="rounded-lg bg-orange-50 p-3 text-center">
                      <p className="text-2xl font-bold text-orange-600">
                        {article.verifications.unverified}
                      </p>
                      <p className="text-xs text-orange-600">待确认</p>
                    </div>
                  </div>
                  
                  <div className="rounded-lg bg-neutral-50 p-4">
                    <p className="text-sm text-neutral-600">
                      整体置信度: <span className="font-bold text-neutral-900">{verificationRate}%</span>
                    </p>
                    <div className="mt-2 h-2 w-full overflow-hidden rounded-full bg-neutral-200">
                      <div
                        className={cn(
                          "h-full rounded-full transition-all",
                          verificationRate >= 80 ? "bg-accent-500" :
                          verificationRate >= 60 ? "bg-yellow-500" : "bg-orange-500"
                        )}
                        style={{ width: `${verificationRate}%` }}
                      />
                    </div>
                  </div>
                </CardContent>
              )}
            </Card>
          </motion.div>
        )}

        {/* 底部操作 */}
        <div className="flex justify-center pt-4">
          <Button
            size="lg"
            variant="secondary"
            onClick={() => router.push("/")}
          >
            <Sparkles className="mr-2 h-4 w-4" />
            生成新文章
          </Button>
        </div>
      </div>
    </div>
  )
}
```
