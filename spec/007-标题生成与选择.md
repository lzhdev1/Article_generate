# Spec-007: 标题生成与选择

## 1. 功能概述

本模块负责：
1. 基于提取的知识和Topic，使用LLM生成5个候选标题
2. 展示标题供用户选择
3. 用户选择一个标题后进入下一步

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| 阿里云百炼 (Qwen-Max) | 标题生成 |
| Celery | 异步任务 |
| Framer Motion | 标题卡片动画 |

---

## 3. 数据模型

### 3.1 标题表

```sql
CREATE TABLE titles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    content VARCHAR(300) NOT NULL,
    reasoning TEXT,           -- 生成理由/风格说明
    style VARCHAR(50),        -- 标题风格
    is_selected BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE INDEX idx_titles_project_id ON titles(project_id);
```

```

```


```
PATCH /api/projects/{project_id}/titles/{title_id}/select
```

**Response 200:**
```json
{
  "success": true,
  "selected_title": "AI重塑教育：2024年个性化学习的五大突破"

```

## 5. 核心代码实现

### 5.1 标题生成服务

```python
# backend/app/services/title_service.py

from typing import List, Dict
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.services.llm_service import llm_service
from app.models.knowledge import ExtractedKnowledge
from app.models.title import Title
from app.models.project import Project

class TitleService:
    
    GENERATE_PROMPT = """你是一位专业的内容编辑，擅长撰写吸引读者的文章标题。

基于以下信息，请生成5个不同风格的文章标题：

## 文章主题
{topic}

## 参考知识

### 关键论点
{key_points}

### 事实数据
{facts}

### 参考文章标题
{source_titles}

## 要求
请生成5个标题，每个标题使用不同风格：
1. 数字列表型：包含具体数字，如"X个方法"、"X大趋势"
2. 问题引导型：以问题形式吸引读者
3. 权威专业型：体现专业深度
4. 情感共鸣型：引发读者情感共鸣
5. 对比冲突型：制造认知冲突或对比

请为每个标题说明生成理由和适用场景。

以JSON格式返回：
{{
  "titles": [
    {{
      "content": "标题内容",
      "reasoning": "生成理由",
      "style": "style_name"
    }}
  ]
}}"""

    async def generate_titles(self, project_id: str, topic: str) -> List[Dict]:
        """生成5个候选标题"""
        async for db in self._get_session():
            # 获取知识库
            results = await db.execute(
                select(ExtractedKnowledge).where(
                    ExtractedKnowledge.project_id == project_id
                )
            )
            knowledge_items = results.scalars().all()
            
            # 整合知识
            all_key_points = []
            all_facts = []
            source_titles = []
            
            for item in knowledge_items:
                for kp in (item.key_points or []):
                    if isinstance(kp, dict):
                        all_key_points.append(kp.get("content", ""))
                    elif isinstance(kp, str):
                        all_key_points.append(kp)
                
                for fact in (item.facts or []):
                    if isinstance(fact, dict):
                        all_facts.append(fact.get("content", ""))
                    elif isinstance(fact, str):
                        all_facts.append(fact)
                
                source_titles.append(item.source_title or "")
            
            # 构建prompt
            prompt = self.GENERATE_PROMPT.format(
                topic=topic,
                key_points="\n".join(f"- {kp}" for kp in all_key_points[:15]),
                facts="\n".join(f"- {f}" for f in all_facts[:10]),
                source_titles="\n".join(f"- {t}" for t in source_titles[:10])
            )
            
            # 调用LLM
            response = await llm_service.chat(prompt)
            result = llm_service.parse_json(response)
            
            return result.get("titles", [])
    
    async def save_titles(self, project_id: str, titles_data: List[Dict]) -> List[Title]:
        """保存标题到数据库"""
        async for db in self._get_session():
            saved_titles = []
            for title_data in titles_data:
                title = Title(
                    project_id=project_id,
                    content=title_data["content"],
                    reasoning=title_data.get("reasoning", ""),
                    style=title_data.get("style", "general")
                )
                db.add(title)
                saved_titles.append(title)
            
            await db.commit()
            for t in saved_titles:
                await db.refresh(t)
            
            return saved_titles
    
    async def select_title(self, project_id: str, title_id: str) -> Title:
        """选择标题"""
        async for db in self._get_session():
            # 取消所有已选择
            results = await db.execute(
                select(Title).where(Title.project_id == project_id)
            )
            for t in results.scalars().all():
                t.is_selected = False
            
            # 设置新选择
            title = await db.get(Title, title_id)
            if not title:
                raise ValueError("Title not found")
            
            title.is_selected = True
            
            # 更新项目状态
            project = await db.get(Project, project_id)
            project.status = "title_selected"
            
            await db.commit()
            return title
    
    async def _get_session(self):
        from app.database import async_session_factory
        return async_session_factory()

title_service = TitleService()
```

### 5.2 标题生成任务

```python
# backend/app/tasks/title_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.task import Task
from app.models.project import Project
from app.services.title_service import title_service

@celery.task(bind=True, name="app.tasks.generate_titles")
def generate_titles_task(self, project_id: str):
    return asyncio.run(_generate_titles_async(self, project_id))

async def _generate_titles_async(celery_task, project_id: str):
    async for db in async_session_factory():
        try:
            project = await db.get(Project, project_id)
            task = await _get_or_create_task(db, project_id, "title_generate", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            await _update_progress(db, task, 30, "正在分析知识并生成标题...")
            
            # 生成标题
            titles_data = await title_service.generate_titles(project_id, project.topic)
            
            await _update_progress(db, task, 70, "正在保存标题...")
            
            # 保存标题
            titles = await title_service.save_titles(project_id, titles_data)
            
            project.status = "title_generating_completed"
            await db.commit()
            
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = {"count": len(titles)}
            await db.commit()
            
            return task.result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

async def _get_or_create_task(db, project_id, task_type, celery_task_id):
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    if not task:
        task = Task(project_id=project_id, task_type=task_type, celery_task_id=celery_task_id)
        db.add(task)
        await db.flush()
    return task

async def _update_progress(db, task, progress, message=None):
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```

### 6.2 前端组件

```tsx
// frontend/src/app/project/[id]/titles/page.tsx

"use client"

import { useEffect, useState } from "react"
import { useParams, useRouter } from "next/navigation"
import { motion, AnimatePresence } from "framer-motion"
import { Check, Sparkles, Loader2 } from "lucide-react"

import { Button } from "@/components/ui/button"
import { Card, CardContent } from "@/components/ui/card"
import { StepIndicator } from "@/components/shared/step-indicator"
import { LoadingSpinner } from "@/components/shared/loading-spinner"
import { api } from "@/services/api"

interface Title {
  id: string
  content: string
  reasoning: string
  style: string
  is_selected: boolean
}

const styleLabels: Record<string, string> = {
  trend_list: "数字列表型",
  question: "问题引导型",
  authority: "权威专业型",
  emotional: "情感共鸣型",
  contrast: "对比冲突型",
  general: "通用型",
}

export default function TitlesPage() {
  const params = useParams()
  const router = useRouter()
  const [titles, setTitles] = useState<Title[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [isGenerating, setIsGenerating] = useState(false)
  const [selectingId, setSelectingId] = useState<string | null>(null)

  useEffect(() => {
    loadTitles()
  }, [])

  const loadTitles = async () => {
    try {
      const data = await api.titles.list(params.id as string)
      if (data.titles.length > 0) {
        setTitles(data.titles)
      } else {
        // 需要先生成标题
        await generateTitles()
      }
    } catch (error) {
      console.error(error)
    } finally {
      setIsLoading(false)
    }
  }

  const generateTitles = async () => {
    setIsGenerating(true)
    try {
      await api.titles.generate(params.id as string)
      // 轮询等待生成完成
      await pollUntilComplete()
      const data = await api.titles.list(params.id as string)
      setTitles(data.titles)
    } catch (error) {
      console.error(error)
    } finally {
      setIsGenerating(false)
    }
  }

  const pollUntilComplete = async () => {
    // 简单轮询实现
    return new Promise<void>((resolve) => {
      const interval = setInterval(async () => {
        const data = await api.titles.list(params.id as string)
        if (data.titles.length > 0) {
          clearInterval(interval)
          resolve()
        }
      }, 2000)
    })
  }

  const handleSelect = async (titleId: string) => {
    setSelectingId(titleId)
    try {
      await api.titles.select(params.id as string, titleId)
      setTitles(titles.map(t => ({
        ...t,
        is_selected: t.id === titleId
      })))
      // 短暂延迟后跳转
      setTimeout(() => {
        router.push(`/project/${params.id}/outline-config`)
      }, 500)
    } catch (error) {
      console.error(error)
      setSelectingId(null)
    }
  }

  if (isLoading) {
    return (
      <div className="flex min-h-screen items-center justify-center">
        <LoadingSpinner size="lg" />
      </div>
    )
  }

  const steps = [
    { id: "search", label: "搜索", status: "completed" },
    { id: "filter", label: "过滤", status: "completed" },
    { id: "extract", label: "提取", status: "completed" },
    { id: "title", label: "标题", status: "current" },
    { id: "outline", label: "大纲", status: "upcoming" },
    { id: "config", label: "配置", status: "upcoming" },
    { id: "generate", label: "生成", status: "upcoming" },
  ]

  return (
    <div className="min-h-screen bg-background">
      <div className="mx-auto max-w-4xl px-4 py-8">
        <StepIndicator steps={steps} className="mb-8" />
        
        <div className="mb-8 text-center">
          <h1 className="text-3xl font-bold">请选择一个标题</h1>
          <p className="mt-2 text-neutral-500">
            基于已提取的知识，为您生成了{titles.length}个候选标题
          </p>
        </div>

        {isGenerating ? (
          <div className="flex flex-col items-center justify-center py-20">
            <LoadingSpinner size="lg" />
            <p className="mt-4 text-neutral-500">正在生成标题...</p>
          </div>
        ) : (
          <div className="space-y-4">
            {titles.map((title, index) => (
              <motion.div
                key={title.id}
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: index * 0.1 }}
              >
                <Card
                  className={`transition-all duration-300 ${
                    title.is_selected
                      ? "border-primary-500 bg-primary-50 shadow-glow-primary"
                      : "hover:border-primary-200 hover:shadow-medium"
                  }`}
                >
                  <CardContent className="p-6">
                    <div className="flex items-start justify-between">
                      <div className="flex-1">
                        <div className="flex items-center gap-3">
                          <span className="flex h-8 w-8 items-center justify-center rounded-full bg-primary-100 text-sm font-bold text-primary-600">
                            {index + 1}
                          </span>
                          <h3 className="text-xl font-semibold text-neutral-900">
                            {title.content}
                          </h3>
                          {title.is_selected && (
                            <Check className="h-5 w-5 text-accent-500" />
                          )}
                        </div>
                        
                        <div className="mt-3 ml-11">
                          <span className="inline-block rounded-full bg-neutral-100 px-3 py-1 text-xs font-medium text-neutral-600">
                            {styleLabels[title.style] || title.style}
                          </span>
                          <p className="mt-2 text-sm text-neutral-500">
                            {title.reasoning}
                          </p>
                        </div>
                      </div>
                      
                      {!title.is_selected && (
                        <Button
                          variant="outline"
                          onClick={() => handleSelect(title.id)}
                          disabled={selectingId !== null}
                          className="ml-4 shrink-0"
                        >
                          {selectingId === title.id ? (
                            <Loader2 className="h-4 w-4 animate-spin" />
                          ) : (
                            "选择此标题"
                          )}
                        </Button>
                      )}
                    </div>
                  </CardContent>
                </Card>
              </motion.div>
            ))}
          </div>
        )}
      </div>
    </div>
  )
}
```
