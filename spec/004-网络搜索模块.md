# Spec-004: 网络搜索模块

## 1. 功能概述

本模块负责根据用户输入的Topic，调用Bing Search API进行网络检索，获取相关内容。支持：
- 关键词提取与优化
- 多轮搜索（不同关键词组合）
- 搜索结果爬取（获取全文内容）
- 搜索结果存储

---

## 2. 技术选型

| 技术 | 用途 |
|------|------|
| Bing Search API v7 | 网络搜索引擎 |
| httpx | 异步HTTP客户端 |
| BeautifulSoup4 | HTML内容提取 |
| readability-lxml | 正文提取 |
| Celery | 异步任务处理 |
| Redis | 任务队列 |

---

## 3. 流程设计

```
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│   接收Topic     │────▶│   关键词优化     │────▶│  调用Bing API   │
│                 │     │ (LLM辅助)        │     │                 │
└─────────────────┘     └─────────────────┘     └─────────────────┘
                                                         │
                                                         ▼
┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│   存储结果      │◀────│   内容爬取      │◀────│   结果解析      │
│   (数据库)      │     │   (全文获取)     │     │                 │
└─────────────────┘     └─────────────────┘     └─────────────────┘
```

### 3.1 搜索策略

1. **关键词生成**：使用LLM从Topic生成3-5个搜索关键词组合
2. **多轮搜索**：每组关键词搜索10条结果
3. **去重**：按URL去重
4. **内容爬取**：获取搜索结果页面的全文内容
5. **结果限制**：最终保留20-30条高质量结果

## 6. 核心代码实现

### 6.1 Bing搜索服务

```python
# backend/app/services/search_service.py

import httpx
from typing import List, Optional
from bs4 import BeautifulSoup
from readability import Document
import asyncio

from app.config import settings
from app.models.search import SearchItem
from app.services.llm_service import llm_service

class SearchService:
    def __init__(self):
        self.api_key = settings.BING_SEARCH_API_KEY
        self.endpoint = settings.BING_SEARCH_ENDPOINT
        self.headers = {"Ocp-Apim-Subscription-Key": self.api_key}
    
    async def generate_search_keywords(self, topic: str) -> List[str]:
        """使用LLM生成搜索关键词"""
        prompt = f"""请根据以下主题生成5个不同的搜索关键词组合，用于在搜索引擎中搜索相关内容。
每个关键词应该从不同角度切入，覆盖主题的各个方面。

主题: {topic}

请以JSON数组格式返回5个关键词："""

        response = await llm_service.chat(prompt)
        # 解析LLM响应获取关键词列表
        keywords = llm_service.parse_json_array(response)
        return keywords[:5]
    
    async def search(self, query: str, count: int = 10) -> List[dict]:
        """调用Bing Search API"""
        params = {
            "q": query,
            "count": count,
            "mkt": "zh-CN",
            "responseFilter": "Webpages",
        }
        
        async with httpx.AsyncClient() as client:
            response = await client.get(
                self.endpoint,
                headers=self.headers,
                params=params,
                timeout=30.0
            )
            response.raise_for_status()
            data = response.json()
        
        results = []
        if "webPages" in data and "value" in data["webPages"]:
            for item in data["webPages"]["value"]:
                results.append({
                    "title": item.get("name", ""),
                    "url": item.get("url", ""),
                    "snippet": item.get("snippet", ""),
                    "source": self._extract_domain(item.get("url", "")),
                    "published_date": item.get("dateLastCrawled", "")[:10] if item.get("dateLastCrawled") else None,
                })
        
        return results
    
    async def crawl_content(self, url: str) -> Optional[str]:
        """爬取网页全文内容"""
        try:
            async with httpx.AsyncClient(follow_redirects=True) as client:
                response = await client.get(
                    url,
                    timeout=15.0,
                    headers={"User-Agent": "Mozilla/5.0 (compatible; ArticleGenerator/1.0)"}
                )
                response.raise_for_status()
                html = response.text
            
            # 使用readability提取正文
            doc = Document(html)
            content = doc.summary()
            
            # 使用BeautifulSoup提取纯文本
            soup = BeautifulSoup(content, "html.parser")
            text = soup.get_text(separator="\n", strip=True)
            
            # 过滤过短的内容
            if len(text) < 200:
                return None
            
            return text[:10000]  # 限制最大长度
            
        except Exception as e:
            print(f"Failed to crawl {url}: {e}")
            return None
    
    def _extract_domain(self, url: str) -> str:
        """从URL提取域名"""
        from urllib.parse import urlparse
        parsed = urlparse(url)
        return parsed.netloc.replace("www.", "")

search_service = SearchService()
```

### 6.2 搜索任务实现

```python
# backend/app/tasks/search_tasks.py

import asyncio
from datetime import datetime
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.tasks.celery_app import celery
from app.database import async_session_factory
from app.models.search import SearchItem
from app.models.task import Task
from app.models.project import Project
from app.services.search_service import search_service

@celery.task(bind=True, name="app.tasks.search")
def search_task(self, project_id: str):
    """执行搜索任务"""
    return asyncio.run(_search_task_async(self, project_id))

async def _search_task_async(celery_task, project_id: str):
    async for db in async_session_factory():
        try:
            # 获取项目信息
            project = await db.get(Project, project_id)
            if not project:
                raise ValueError(f"Project {project_id} not found")
            
            # 更新任务状态
            task = await _get_or_create_task(db, project_id, "search", celery_task.request.id)
            task.status = "running"
            task.started_at = datetime.utcnow()
            await db.commit()
            
            # 步骤1: 生成搜索关键词
            await _update_progress(db, task, 10, "正在生成搜索关键词...")
            keywords = await search_service.generate_search_keywords(project.topic)
            
            all_results = []
            seen_urls = set()
            
            # 步骤2: 执行搜索
            total_keywords = len(keywords)
            for i, keyword in enumerate(keywords):
                progress = 10 + int((i / total_keywords) * 60)
                await _update_progress(db, task, progress, f"正在搜索: {keyword} ({i+1}/{total_keywords})")
                
                results = await search_service.search(keyword, count=10)
                
                for result in results:
                    if result["url"] not in seen_urls:
                        seen_urls.add(result["url"])
                        all_results.append(result)
                
                # 控制请求频率
                await asyncio.sleep(0.5)
            
            # 步骤3: 爬取内容
            await _update_progress(db, task, 75, "正在获取文章内容...")
            
            crawled_items = []
            for i, result in enumerate(all_results[:30]):  # 最多爬取30条
                progress = 75 + int((i / min(30, len(all_results))) * 20)
                await _update_progress(db, task, progress, f"正在获取内容 ({i+1}/{min(30, len(all_results))})")
                
                content = await search_service.crawl_content(result["url"])
                
                # 创建搜索项
                search_item = SearchItem(
                    project_id=project_id,
                    title=result["title"],
                    url=result["url"],
                    snippet=result["snippet"],
                    source=result["source"],
                    published_date=result["published_date"],
                    crawled_content=content
                )
                db.add(search_item)
                crawled_items.append(search_item)
                
                await asyncio.sleep(0.3)  # 控制请求频率
            
            await db.commit()
            
            # 更新项目状态
            project.status = "searching_completed"
            await db.commit()
            
            # 完成任务
            task.status = "completed"
            task.progress = 100
            task.completed_at = datetime.utcnow()
            task.result = {
                "total_results": len(crawled_items),
                "keywords_used": keywords,
                "with_content": sum(1 for item in crawled_items if item.crawled_content),
            }
            await db.commit()
            
            return task.result
            
        except Exception as e:
            await db.rollback()
            task.status = "failed"
            task.error = str(e)
            await db.commit()
            raise

async def _get_or_create_task(db: AsyncSession, project_id: str, task_type: str, celery_task_id: str) -> Task:
    """获取或创建任务记录"""
    result = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == task_type)
    )
    task = result.scalar_one_or_none()
    
    if not task:
        task = Task(
            project_id=project_id,
            task_type=task_type,
            celery_task_id=celery_task_id
        )
        db.add(task)
        await db.flush()
    
    return task

async def _update_progress(db: AsyncSession, task: Task, progress: int, message: str = None):
    """更新任务进度"""
    task.progress = progress
    if message:
        task.result = {"message": message}
    await db.commit()
```

### 6.3 API端点实现

```python
# backend/app/api/endpoints/search.py

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

from app.database import get_db
from app.models.search import SearchItem
from app.tasks.search_tasks import search_task

router = APIRouter()

@router.post("/projects/{project_id}/search", status_code=status.HTTP_202_ACCEPTED)
async def start_search(
    project_id: str,
    db: AsyncSession = Depends(get_db)
):
    """启动搜索任务"""
    # 验证项目存在
    project = await db.get(Project, project_id)
    if not project:
        raise HTTPException(status_code=404, detail="Project not found")
    
    # 检查是否已有搜索任务
    existing_task = await db.execute(
        select(Task).where(Task.project_id == project_id, Task.task_type == "search")
    )
    task = existing_task.scalar_one_or_none()
    
    if task and task.status == "running":
        return {
            "task_id": str(task.id),
            "status": "running",
            "message": "搜索任务正在执行中"
        }
    
    # 启动异步任务
    celery_result = search_task.delay(project_id)
    
    return {
        "task_id": str(celery_result.id),
        "status": "pending",
        "message": "搜索任务已创建"
    }

@router.get("/projects/{project_id}/search")
async def get_search_results(
    project_id: str,
    db: AsyncSession = Depends(get_db)
):
    """获取搜索结果"""
    results = await db.execute(
        select(SearchItem)
        .where(SearchItem.project_id == project_id)
        .order_by(SearchItem.created_at)
    )
    items = results.scalars().all()
    
    return {
        "items": [
            {
                "id": str(item.id),
                "title": item.title,
                "url": item.url,
                "snippet": item.snippet,
                "source": item.source,
                "published_date": item.published_date.isoformat() if item.published_date else None,
                "crawled_content": item.crawled_content,
                "is_relevant": item.is_relevant,
                "relevance_score": item.relevance_score,
            }
            for item in items
        ],
        "total": len(items)
    }
```

## 7. WebSocket实时推送

```python
# backend/app/api/websocket.py

from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from typing import Dict
import asyncio
import json

router = APIRouter()

# 连接管理器
class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, WebSocket] = {}

    async def connect(self, project_id: str, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[project_id] = websocket

    def disconnect(self, project_id: str):
        if project_id in self.active_connections:
            del self.active_connections[project_id]

    async def send_progress(self, project_id: str, data: dict):
        if project_id in self.active_connections:
            await self.active_connections[project_id].send_json(data)

manager = ConnectionManager()

@router.websocket("/ws/tasks/{project_id}")
async def websocket_endpoint(websocket: WebSocket, project_id: str):
    await manager.connect(project_id, websocket)
    try:
        while True:
            # 保持连接，接收客户端心跳
            data = await websocket.receive_text()
            if data == "ping":
                await websocket.send_json({"type": "pong"})
    except WebSocketDisconnect:
        manager.disconnect(project_id)

# 在任务中发送进度更新
async def notify_progress(project_id: str, progress: int, message: str):
    await manager.send_progress(project_id, {
        "type": "progress",
        "progress": progress,
        "message": message
    })
```

## 9. 错误处理

| 错误类型 | 处理方式 |
|----------|----------|
| API Key无效 | 记录日志，标记任务失败 |
| 请求超时 | 重试3次，间隔递增 |

| 爬取失败 | 跳过该条结果，继续处理 |
